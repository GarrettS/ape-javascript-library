 32 	/** 
 33 	 * creates a namespace split on "."
 34 	 * does <em>not</em> automatically add APE to the front of the chain, as YUI does.
 35 	 * @param {String} s the namespace. "foo.bar" would create a namespace foo.bar, but only
 36 	 * if that namespace did not exists.
 37 	 */
 38 	namespace : function(s) {
 39 		var packages = s.split(".");
 40 		var pkg = window;
 41 		for (var i = 0; i < packages.length; i++) {
 42 			var name = packages[i];
 43 
 44 			// Internet Explorer does not support 
 45 			// hasOwnProperty on things like window, so call Object.prototype.hasOwnProperty.
 46 			// Opera does not support the global object or [[Put]] properly (see below)
 47 			if(!Object.prototype.hasOwnProperty.call(pkg, name)) {
 48 				pkg[name] = new Package((pkg.qualifiedName||"APE")+"."+name);
 49 			}
 50 		    pkg = pkg[name];
 51 		}
 52 
 53 		/** @constructor Package
 54 		 * @private 
 55 		 */
 56 		Package.prototype.toString = function(){
 57 				return"["+this.qualifiedName+"]";
 58 			};
 59 		function Package(qualifiedName) {
 60 			this.qualifiedName = qualifiedName;
 61 		}
 62 		return pkg;
 63 	},
 64 	
 65 	/**
 66 	 * Prototype inheritance
 67 	 * @param {Object} subclass 
 68 	 * @param {Object} superclass
 69 	 * @param {Object} mixin If present, <var>mixin</var>'s own properties are copied to receiver 
 70 	  * using APE.mixin(subclass.prototoype, superclass.prototype).
 71 	 * 
 72 	 */
 73 	extend : function(subclass, superclass, mixin) {
 74 		function f(){}
 75 		f.prototype = superclass.prototype;
 76 		subclass.prototype = new f;
 77 		subclass.prototype.constructor = subclass;
 78 		if(typeof mixin == "object")
 79 			APE.mixin(subclass.prototype, mixin);
 80 	},
 81 
 82 	/**
 83 	 * Shallow copy of properties; does not look up prototype chain.
 84 	 * Copies all properties in s to r, using hasOwnProperty.
 85 	 * @param {Object} r the receiver of properties.
 86 	 * @param {Object} s the supplier of properties.
 87 	 */
 88 	mixin : function(r, s) {
 89 		var jscriptSkips = ['toString', 'valueOf'];
 90 		for(var prop in s)
 91 			if(s.hasOwnProperty(prop))
 92 				r[prop] = s[prop];
 93 
 94 		// JScript DontEnum bug.
 95 		for(var i = 0; i < jscriptSkips.length; i++) {
 96 			var skipped = jscriptSkips;
 97 			r[skipped] = s[skipped];
 98 		}
 99 	},
100 
101 	/**
102 	 * explicit intertype declarations.
103 	 * @param {Object} receiver this object recieves properties
104 	 * @param {Object} supplier supplies properties
105 	 * @param {Array} strings of property names to look for in supplier
106 	 * @throws {Error} Error if any property in propsArray is not found in the supplier.
107 	 */
108 	crossCut : function(receiver, supplier, propsArray) {
109 		if(typeof propsArray == "string") propsArray = [propsArray];
110 		for(var i = 0, len = propsArray.length; i < len; i++) {
111 			var p = propsArray[i];
112 			if(! (p in supplier)) {
113 				throw Error(p + " not found in supplier. Did you check the prototype?");
114 			}
115 			receiver[p] = supplier[p];
116 		}
117 	},

121 	/** Creational method meant for being cross-cut.
122 	 * We really need a new.apply()/construct.apply() method, but ES doesn't have it.
123 	 * @param {HTMLElement} el An element that has an id.
124 	 * @requires {Object} an object to bind to.
125 	 * @aspect 
126 	 * @scope {Function} Anything. APE.getByNode is intended to be bouund to a constructor function
127 	 * that accepts an element that has an ID..
128 	 * @return {new this(el)}
129 	 */
130 	getByNode : function(el) {
131 		if(!this.hasOwnProperty("instances")) this.instances = {};
132 		return this.instances.hasOwnProperty(el.id) && this.instances[el.id] || (this.instances[el.id] = new this(el));
133 	},

152 	/** 
153 	 * Wraps o's method with advice.
154 	 * If the advice returns false, o's method won't fire.
155 	 *
156 	 * @param {Object} o the object that has the method.
157 	 * @param {string} funcName o's named method.
158 	 * @param {Function} advice to wrap.
159 	 * If returning false, o's method won't fire;
160 	 * If returning an array, the array is passed to apply to o's method call.
161 	 * Otherwise, just call o with the original arguments it was called with (default).
162 	 */
163 	addBefore : function(o, funcName, advice) {
164 		var m = o[funcName];
165 		o[funcName] = function beforeAdvice() {
166 			var proceed = advice.apply(o, arguments);
167 			// Call o's method.
168 			if(proceed !== false) {
169 
170 				// Default: use arguments that o[funcName] was called with.
171 				var argsToApply = arguments;
172 
173 				// Did we get other arguments?
174 				if(APE.isArray(proceed)) {
175 					argsToApply = proceed;
176 				}
177 				return m.apply(o, argsToApply);
178 			}
179 		}
180 	},
181 
182 	/** 
183 	 * @param {Object} o The object whose method is to be advised.
184 	 * @param {string} funcName o's named method.
185 	 * @param {Function} advice to call after o[funcName] is called.
186 	 *
187 	 * @return the returnValue of the main function for argument [0],
188 	 * plus the arguments to the original function.
189 	 * 
190 	 */
191 	addAfter : function(o, funcName, advice) {
192 		var m = o[funcName];
193 		o[funcName] = function afterAdvice() {
194 			var returnValue = m.apply(o, arguments);
195 
196 			// First argument is the return value from the method.
197 			var args = [returnValue];
198 			args = args.concat.apply(args, arguments);
199 			return advice.apply(o, args);
200 		}
201 	},

227 var STRING_TRIM_EXP = /^\s+|\s+$/g;
228 /** 
229  * @member String
230  * @addon trim
231  * @return {String} trimmed copy of this string's whitespace (removes leading and trailing whitespace).
