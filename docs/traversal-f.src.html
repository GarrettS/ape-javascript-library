1 APE.namespace("APE.dom");
  2 (function(){
  3 
  4     var docEl = document.documentElement;
  5 
  6     APE.mixin(
  7         APE.dom, {
  8         contains : getContains(),
  9         findAncestorWithAttribute : findAncestorWithAttribute,
 10         findAncestorWithTagName : findAncestorWithTagName,
 11         findNextSiblingElement : findNextSiblingElement,
 12         findPreviousSiblingElement : findPreviousSiblingElement,
 13         getChildElements : getChildElements
 14     });
 15 
 16     /** 
 17      * @memberOf APE.dom
 18      * @return {boolean} true if a contains b.
 19      * Internet Explorer's native contains() is different. It will return true for:
 20      * code body.contains(body); 
 21      * Whereas APE.dom.contains will return false.
 22      */
 23     function getContains(){
 24         if('compareDocumentPosition'in docEl)
 25             return function(el, b) {
 26                 return (el.compareDocumentPosition(b) & 16) !== 0;
 27         };
 28         else if(docEl.contains)
 29             return function(el, b) {
 30                 return el !== b && el.contains(b);
 31         };
 32         return function(el, b) {
 33             if(el === b) return false;
 34             while(el != b && (b = b.parentNode) != null);
 35             return el === b;
 36         };
 37     }
 38     /** 
 39      * @memberOf APE.dom
 40      * @param {HTMLElement} el the element to start from.
 41      * @param {String} attName the name of the attribute.
 42      * @param {String} [value] the value of the attribute. If omitted, then only the 
 43      * presence of attribute is checked and the value is anything.
 44      * @return {HTMLElement} closest ancestor with <code>attName</code> matching value.
 45      * Returns null if not found.
 46      */
 47     function findAncestorWithAttribute(el, attName, value) {
 48         for(var map, parent = el.parentNode;parent != null;){
 49             map = parent.attributes;
 50             if(!map) return null;
 51             var att = map[attName];
 52             if(att && att.specified)
 53                 if(att.value === value || (value === undefined))
 54                     return parent;            
 55             parent = parent.parentNode;
 56         }
 57         return null;
 58     }
 59 
 60     var isUpperCase = /^H/.test(docEl.tagName);
 61     function findAncestorWithTagName(el, tagName) {
 62         tagName = (isUpperCase ? tagName.toUpperCase() : tagName.toLowerCase());
 63         for(var parent = el.parentNode;parent != null; ){
 64             if( parent.tagName === tagName )
 65                 return parent;
 66             parent = parent.parentNode;
 67         }
 68         return null;
 69     }
 70 
 71     function findNextSiblingElement(el) {
 72         for(var ns = el.nextSibling; ns != null; ns = ns.nextSibling)
 73             if(ns.tagName) 
 74                 return ns;
 75             return null;
 76     }
 77 
 78     function findPreviousSiblingElement(el) {
 79         for(var ps = el.previousSibling; ps != null; ps = ps.previousSibling)
 80             if(ps.tagName) 
 81                 return ps;
 82             return null;
 83     }

 85     function getChildElements(el) {    
 86         var i = 0, ret = [],
 87             cn = el.childNodes;
 88         for(var len = cn.length; i < len; i++) {
 89             if("tagName"in cn[i])
 90                 ret[ret.length] = cn[i];
 91         }
 92         return ret;
 93     }
