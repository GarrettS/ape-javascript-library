 24 /** Returns a ColorRGB based on a number.
 25  * @param {uint} number from 0 - 0xFFFFFF
 26  */
 27 APE.color.ColorRGB.fromNumber = function(n) {
 28 	return new APE.color.ColorRGB((0xFF0000 & n) >> 16, (0xFF00 & n) >> 8, 0xFF & n);

 31 /** Returns a ColorRGB based on a hex string: 
 32  * #FFFFFF or #FFF
 33  */
 34 APE.color.ColorRGB.fromHexString = function(hex) {
 35 	if(!this.hexPattern.test(hex)) 
 36 		throw Error("ColorRGB.fromHexString(hex) invalid input: " + hex);
 37 	hex = hex.substring(1); // chop "#"
 38 	var n = parseInt(hex, 16);
 39 	var rMask, gMask, bMask;
 40 	var r, g, b;
 41 	var ColorRGB = APE.color.ColorRGB;
 42 
 43 	if(hex.length == 3) {
 44 		r =  n & 0xF00, g = n & 0x0F0, b = n & 0x00F;
 45 		return new ColorRGB( 
 46 			 (r >> 4) + (r >> 8), 
 47 			  g +       (g >> 4), 
 48 			 (b << 4) + b
 49 		);
 50 	}
 51 	return ColorRGB.fromNumber(n);
 52 };

 54 /** Color.RGB.rgbToHex 
 55  * rgbPattern
 56  * this pattern matches a String that looks like this: 
 57  * rgb( + int + "," + int + "," +int + ")"
 58  *
 59  * rgbPattern is used by rgbToHex, declared outside the function for 
 60  * performance -- compiled once by the engine. 
 61  */
 62 APE.color.ColorRGB.hexPattern = /(?:#)[0-9a-z]{3,6}/i;
 63 APE.color.ColorRGB.rgbPattern = /\brgb\(\s*([\d]{1,3})(%)?\s*\,\s*([\d]{1,3})(%)?\s*\,\s*([\d]{1,3})(%)?\s*\)/i;
 64 
 65 APE.color.ColorRGB.fromString = function(inp) {
 66 	var ColorRGB = APE.color.ColorRGB;
 67 	return (ColorRGB.hexPattern.test(inp) ? ColorRGB.fromHexString(inp) : ColorRGB.fromRgbString(inp));
 68 };

 70 /** 
 71  *	@param {String} rgbString - if invalid, 
 72  *  a null ColorRGB object is returned;
 73  */ 
 74 APE.color.ColorRGB.fromRgbString = function(rgbString) {
 75 
 76 	var ColorRGB = APE.color.ColorRGB;
 77 	var rgbPattern = ColorRGB.rgbPattern;
 78 	var rgbArray = [];
 79 	var rgb = rgbString.match(rgbPattern);
 80 	
 81 	if(rgb != null) {
 82 		
 83 		// ignore the zeroeth match of rgb.
 84 		// loop begins at 1 to match first paren.
 85  		for(var j = 1, len = rgb.length; j < len; j++)
 86  			if( isFinite(parseInt(rgb[j])) ) {
 87 				if(j+2 < rgb.length && rgb[j+1] == "%")
 88 					rgbArray[rgbArray.length] = (255 * parseInt(rgb[j])/100);
 89 				else
 90 					rgbArray[rgbArray.length] = parseInt(rgb[j]);
 91 			}
 92 		return new ColorRGB(rgbArray[0], rgbArray[1], rgbArray[2]);
 93 	}
 94 	return new ColorRGB();
 95 };

 97 /** Takes a midpoint color between two colors. Useful for animation. 
 98  * @param rationalValue ratio [0..1] of endColor:startColor color for result.
 99  * @param mixedColor {ColorRGB} - a color can be passed in for reuse (this is useful for
100  * animation, to avoid creation of many objects).
101  *
102  * @return APE.color.ColorRGB new color that is a blend of
103  *
104  * rationalValue/1 new color and 1-rationalValue startColor. 
105  */
106 APE.color.ColorRGB.blend = function(startColor, endColor, rationalValue, mixedColor) {
107 	var inverse = 1-rationalValue;
108 
109 	var r = startColor.r * inverse + endColor.r * rationalValue;
110 	var g = startColor.g * inverse + endColor.g * rationalValue;
111 	var b = startColor.b * inverse + endColor.b * rationalValue;
112 	// Math.abs: (x ^ (x >> 31)) - (x >> 31);
113 	// converts result to int (floor).
114 	r = (r ^ (r >> 31)) - (r >> 31);
115 	g = (g ^ (g >> 31)) - (g >> 31);
116 	b = (b ^ (b >> 31)) - (b >> 31);
117 
118 	if(mixedColor instanceof this) {
119 		mixedColor.r = r;
120 		mixedColor.g = g;
121 		mixedColor.b = b;
122 		return mixedColor;
123 	}
124 	return new this(r, g, b);
125 };

127 APE.color.ColorRGB.prototype = {
128 
129 	r : 0xff, g : 0xff,  b : 0xff,
130 	
131 	/** 
132 	  * @return {ColorHSV} color object.
133 	  */
134 	toHSV : function() {
135 		
136 		var max = Math.max(this.r, this.g, this.b);
137 		var min = Math.min(this.r, this.g, this.b);
138 		
139 		var s = 0;
140 		var h = 0;
141 		
142 		if(max > min) {
143 			switch(max) {
144 				case this.r :
145 					h = (this.g - this.b) / (max-min);
146 					break;
147 				case this.g :
148 					h = 2 + ((this.b - this.r) / (max-min));
149 					break;
150 				case this.b :
151 					h = 4 + ((this.r - this.g) / (max-min));
152 					break;
153 			}
154 			s = (max - min) / max;
155 		}
156 		h *= 60;
157 		if(h < 0) 
158 			h += 360;
159 		
160 		return new APE.color.ColorHSV(h, s, max/255);
161 	},
162 	
163 	toString : function() {
164 		return "rgb(" + this.r
165 				+ ", " + this.g 
166 				+ ", " + this.b + ")";
167 	},
168 	
169 	toHexString : function() {
170 		return "#" + toHexByte(this.r) + toHexByte(this.g) + toHexByte(this.b);
171 		
172 		function toHexByte(bite) {
173 			var hex = bite.toString(16);
174 			return (hex.length == 2 ? hex : "0" + hex);
175 		}
176 	},
177 	
178 	equals : function(c) {
179 		return (this.r == c.r) && (this.b == c.b) && (this.g == c.g);
180 	},
181 	
182 	isValid : function() {
183 		return validComponent(this.r) && validComponent(this.g) && validComponent(this.b);
184 		function validComponent(c) {
185 			return isFinite(c) && c >= 0 && c <= 255;
186 		}
187 	},
188 
189 	valueOf : function() {
190 		return (this.r << 16) + (this.g << 8) + this.b;
191 	}
192 };

212 APE.color.ColorHSV.prototype = {
213 	
214 	h : 360, s: 1.0, v : 1.0,
215 	
216 	toRGB : function() {
217 		var H = this.h/360, 
218 			S = this.s, 
219 			V = this.v;
220 		
221 		var R, G, B;
222 		
223 		if ( S == 0 ) {    //HSV values = From 0 to 1
224 			R = G = B = V;
225 		}
226 		else {
227 			H *= 6;
228 
229 			// Math.floor: Right shift by 0, faster than Math.floor.
230 			var i = H >> 0; 
231 			var var1 = V * ( 1 - S )
232 			var var2 = V * ( 1 - S * ( H - i ) )
233 			var var3 = V * ( 1 - S * ( 1 - ( H - i ) ) )
234 		
235 			switch(i) {
236 				case 0 : 
237 					R = V; G = var3 ; B = var1;
238 					break;
239 				case 1 :
240 					R = var2; G = V; B = var1;
241 					break;
242 				case 2 :
243 					R = var1 ; G = V; B = var3;
244 					break;
245 				case 3 :
246 					R = var1 ; G = var2 ; B = V;
247 					break;
248 				case 4 :
249 					R = var3 ; G = var1 ; B = V;
250 					break;
251 				default : 
252 					R = V; G = var1 ; B = var2;
253 			}
254 		}
255 		return new APE.color.ColorRGB( R*255 >> 0, G*255 >> 0, B*255 >> 0);
256 	},
257 	
258 	toString : function() {
259 		return "[ " + this.h.toFixed(0)
260 				+ ", " + this.s.toFixed(2) 
261 				+ ", " + this.v.toFixed(2) + " ]";
262 	}
263 };
