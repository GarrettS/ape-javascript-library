 39 /**  
 40  * @constructor Animation
 41  * @param {ufloat} [duration] Number of seconds to run the animation (default is 1).
 42  */
 43 APE.anim.Animation = function( duration ) {
 44 	if(typeof duration == "number")
 45 		this.duration = duration * 1000; // default 1 sec.
 46 	this.timeLimit = this.duration; // for SeekTo()
 47 };

 49 /** 
 50  * Calculates the distance between two points.
 51  * @param {x:int, y:int} start coords
 52  * @param {x:int, y:int} end coords
 53  * @static
 54  */
 55 APE.anim.calculateHypoteneuse = function(start, end) {
 56 	return Math.ceil(Math.sqrt(Math.pow(start.x - end.x, 2) + Math.pow(start.y - end.y, 2)));
 57 };

 59 /** 
 60  * blendValue is useful for transitioning between two numeric values.
 61  * @param v1 the first value 
 62  * @param v2 other value to mix in
 63  * @param position - ratio blend of v2:v1
 64  * @static
 65  */
 66 APE.anim.Animation.blendValue = function(v1, v2, position) {
 67 	var inverse = 1 - position;
 68 	return (v1 * inverse) + (v2 * position);
 69 };

 71 APE.anim.Animation.prototype = {
 72 	
 73 	paused : false,
 74 
 75 	/** {Number} duration of how long the animation will run (milliseconds). */
 76 	duration : 1000,
 77 
 78 	/** {Number} duration of how long the animation will run.
 79 	 * Used internally for seekTo().
 80 	 */
 81 	timeLimit : 1000,
 82 
 83 	isReversed : false,
 84 
 85 	startOffset : 0,
 86 	endOffset : 1,
 87 
 88 	position : 0,
 89 
 90 	/** @event */
 91 	onstart : function(){},
 92 	/** @event */
 93 	onend : function(){},
 94 	
 95 	/** if an error occurred, onabort throws the error.
 96 	 * this can be overridden (shadowed) by adding 
 97 	 * an onabort() to the Animation instance.
 98 	 * e.g. APE.EventRegistry.add( myAnim, "onabort", myErrorHandler );
 99 	 * @param {Error} ex the error that occured.
100 	 */
101 	onabort : function(ex) {
102 		throw ex;
103 	},
104 
105 	/** Run must be implemented by user.
106 	 * @param position {Number} 0-1
107 	 * 
108 	 * Implementation generally looks something like:<pre>
109 	 * anim.run = function( position ) {
110 	 *   document.body.style.borderWidth = ( 12 * position ) + "px"
111 	 * };</pre>
112 	 */
113 	run : function() { throw Error("APE.anim: " + this.id + " run() not implemented."); },
114 
115 	/** 
116 	 * Call this method to start the anim.
117 	 */
118 	start : function() {
119 		if(this.paused) return;
120 		this.playing = true;
121 		this.timeLimit = this.duration;
122 		this._start();
123 	},
124 
125 	/**
126 	 *  timeLimit is not calculated here.
127 	 * @private
128 	 */
129 	_start : function() {
130 		// Unregister the animation before setting startTime.
131 		APE.anim.Manager.unregister(this);
132 		this._startTime = new Date-0;
133 		this.onstart();
134 		APE.anim.Manager.register(this);
135 		this.started = true;
136 	},
137 
138 	/** 
139 	 * Seeks to a certain position along the
140 	 * duration timeline. 
141 	 * @param pos {ufloat} [0-1]. 
142 	 */
143 	seekTo : function(pos) {
144 
145 		// The new distance is the difference between the 
146 		// desired position (pos) and the currentPosition (position).
147 		this.startOffset = this.position;
148 		this.endOffset = pos;
149 		var distance = Math.abs(pos - this.startOffset);
150 
151 		// The new timeLimit is a percentage of the the full duration.
152 		this.timeLimit = this.duration * distance;
153 
154 		this._startTime = new Date-0;
155 		this.onstart();
156 
157 		APE.anim.Manager.unregister(this);
158 		this._start();
159 		if(!this.started) { 
160 			this.start();
161 			this.isReversed = false;
162 			return;
163 		}
164 		this.isReversed = (pos < this.position);
165 			
166 		APE.anim.Manager.register(this);
167 		this.started = true;
168 	},
169 	
170 	/** 
171 	 * resets the animation to position 0.
172 	 */
173 	reset : function() {
174 		this.position = 0;
175 		this.timeLimit = this.duration;
176 	},
177 	
178 	/** 
179 	 * pauses the animation.
180 	 */
181 	pause : function() {
182 		this.paused = true;
183 		this.elapsedTime = new Date-this._startTime;
184 		APE.anim.Manager.unregister(this);
185 	},
186 	
187 	/** 
188 	 * unpauses the animation.
189 	 */
190 	resume : function() {
191 		this.paused = false;
192 		// Pick up from last time frame.
193 		this._startTime = new Date - this.elapsedTime;
194 		APE.anim.Manager.register(this);
195 	},
196 	
197 	/** 
198 	 * Called by the anim.Manager
199 	 * @private
200 	 */
201 	_playFrame : function() {
202 
203 		var elapsed = new Date - this._startTime;
204 		if(elapsed >= this.timeLimit) {
205 			this.run(this.endOffset);
206 			this._end();
207 			return;
208 		}
209 		var distanceTraveled = (elapsed / this.duration);
210 		if(this.isReversed) {
211 			distanceTraveled = -distanceTraveled;
212 		}
213 		this.position = this.startOffset + distanceTraveled;
214 		// If an error occurs, cancel the APE.anim and throw the error.
215 		try {
216 			this.run( this.position );
217 		} 
218 		catch(ex) {
219 			// If an error occurs, abort the anim.
220 			this.abort(ex);
221 			this._end(false);
222 		}
223 	},
224 
225 	toString : function() {
226 		return"Animation {duration millis: " + this.duration + 
227 			", position:" + this.position+"}";
228 	},
229 
230 	/** Ends the anim.
231 	 * @param {boolean} ended, if true, calls onend().
232 	 * Note, this is not a pause() method.
233 	 */
234 	stop : function(ended) {
235 		this._end(ended);
236 	},
237 
238 	/** Cancels the anim where it is; does not call onend()
239 	 */
240 	abort : function(ex) {
241 		this._end(false);
242 		this.onabort(ex||{});
243 	},
244 	
245 	_end : function(complete) {
246 		APE.anim.Manager.unregister(this);
247 		if(complete !== false) {
248 			this.onend();
249 		}
250 	}
251 };

254 /** 
255  * @class 
256  * @protected - for internal use.
257  * Calls _playFrame() on each anim.
258  * This object is for internal use, tightly coupled to Animation.
259  */
260 APE.anim.Manager = new function() {
261 	
262 	/** 
263 	 * doubleThreaded animations exhibit varying 
264 	 * degrees of improvement on performance, depending on the browser.
265 	 * Particularly, any version of Safari on Mac or Windows, and 
266 	 * Internet Explorer on Win9.x
267 	 * Set this flag to false to reduce CPU hogging.
268 	 */
269 	this.isDoubleThreaded = true;
270 
271 	/** 
272 	 * @private - keeps track of activeAnimations in main loop.
273 	 */
274 	var activeAnimations = [];
275 
276 	/** 
277 	 * @private - thread timers.
278 	 */
279 	var intervalId, intervalId2;
280 
281    /** 
282     * Registers the animation in the thread pool once.
283 	* @param {Animation} anim the animation to register.
284 	*/
285 	this.register = function(anim) {
286 		if(activeAnimations.length === 0)
287 			start();
288 		for(var i = 0; i < activeAnimations.length; i++) {
289 			if(activeAnimations[i] === anim) {
290 				return;
291 			}
292 		}
293 		activeAnimations.push(anim);
294 	};
295 	
296    /** 
297     * Registers the animation in the thread pool once.
298 	* @param {Animation} anim the animation to register.
299 	*/
300 	this.unregister = function(anim) {
301 		for(var i = 0; i < activeAnimations.length; i++) {
302 			if(activeAnimations[i] === anim) {
303 
304 				activeAnimations.splice(i, 1);
305 			}
306 		}
307 		if(activeAnimations.length == 0) {
308 			activeAnimations = [];
309 			stop();
310 		}
311 	};
312 
313 	this.toString = function() {
314 		return"APE anim.Manager";
315 	};
316 	
317 	/** 
318 	 * starts run();
319 	 * @private
320 	 */
321 	function start() {
322 		// Double threaded.
323 		intervalId = window.setInterval(run, 20);
324 		if(this.isDoubleThreaded)
325 			this.intervalId2 = window.setInterval(run, 45);
326  	}
327 
328 	/** 
329 	 * Plays the frame for each animation.
330 	 * @private - starts run().
331 	 */
332 	function run() {
333 		// Check animations.legth each iteration. Some animations 
334 		// finish before others and must be unregistered.
335 		for(var i = 0; i < activeAnimations.length; i++)
336 			activeAnimations[i]._playFrame();
337 	}
338 
339 	/** 
340 	 * Called automatically when there are no more activeAnimations to run.
341 	 */
342 	function stop() {
343 		window.clearInterval(intervalId);
344 		window.clearInterval(intervalId2);
345 	}
346 };

348 /**

350  * Easing functions.

352 APE.anim.Transitions = {
353 
354 	accel: function (pos) {
355     	return pos*pos*pos;
356     },
357 
358 	decel : function(pos) {
359 		pos = 1-pos;
360     	return 1-(pos*pos*pos);
361 	},
362 	
363 	/**
364 	 * For better performance, use (1-position) instead.
365 	 */
366     reverse: function(pos) {
367 		return 1-pos;
368     },
369 
370 	// need pulse and throb... 
371 	// pulse(pos, times) -> accel, then drop quickly to 0, n times
372 	// throb(pos, times) -> jump quickly to 1, then decel to 0, n times.
373 	// Type of parameterized sinoidal effect (pos, accelTo, delelFrom);
374 
375     /** Plays the APE.anim forwards and backwards 
376 	 *  in the same duration.
377 	 */
378 	loop : function(pos) {
379 		// If almost at 1/2, move to 1, else move back.
380 		return pos <= 1/2 ? (pos > 3/7 ? 1 : 2 * pos) : 1 - (2 * -((1/2) - pos));
381 	},
382 
383 	/**
384 	 * sinoidal and spring come from Scriptaculous.
385 	 */
386     sinoidal: function(pos) {
387 		return (-Math.cos(pos*Math.PI)/2) + 0.5;
388     },
389 
390     spring : function(pos) { 
391 		return 1 - (Math.cos(pos * 4.5 * Math.PI) * Math.exp(-pos * 6)); 
392     },
393 	
394 	/**
395 	 * Based on Easing Equations v2.0 
396 	 * (c) 2003 Robert Penner, all rights reserved. 
397 	 * This work is subject to the terms in http://www.robertpenner.com/easing_terms_of_use.html
398 	 * Adapted for Scriptaculous by Ken Snyder (kendsnyder ~at~ gmail ~dot~ com) June 2006 
399 	 */
400 	swingTo : function(pos) { 
401 		var s = 1.70158; 
402 		return (pos-=1)*pos*((s+1)*pos + s) + 1;
403 	},
404 
405 	swingToFrom : function(pos) {
406 		var s = 1.70158; 
407 		if ((pos/=0.5) < 1) return 0.5*(pos*pos*(((s*=(1.525))+1)*pos - s)); 
408 		return 0.5*((pos-=2)*pos*(((s*=(1.525))+1)*pos + s) + 2); 
409 	},
410 
411 	toString : function() {
412 		return"APE anim Transitions";
413 	}
