1 /**
  2  * @author Garret Smith
  3  */
  4 APE.namespace("APE.dom");
  5 
  6 (function() {
  7     /**@name APE.dom*/
  8 
  9     var docEl = document.documentElement,
 10         IS_BODY_ACTING_ROOT = docEl && docEl.clientWidth === 0;
 11 	
 12 	var getScrollOffsets = ("pageXOffset"in window ? function() {
 13 		return{ left:window.pageXOffset, top: window.pageYOffset};
 14 	} : IS_BODY_ACTING_ROOT ? function() {
 15 		var body = document.body;
 16 		return{ left: body.scrollLeft, top : body.scrollTop};
 17 	} :
 18 		function() {
 19 		return{ left : docEl.scrollLeft, top : docEl.scrollTop };
 20 	}),
 21 
 22     /** @memberOf APE.dom 
 23      * @return an object with <code>width</code> and <code>height</code>.
 24 	 * This will exhibit a bug in Mozilla, which is often 5-7 pixels off.
 25      */
 26 	getViewportDimensions = (typeof document.width == "number" ? function(){
 27 		return{ width : Math.min(document.width, docEl.clientWidth), 
 28 			height : Math.min(document.height, docEl.clientHeight)
 29 		};
 30 	} :	"innerWidth"in window ? function() {
 31 			return { width: window.innerWidth, height: window.innerHeight };
 32 	}:IS_BODY_ACTING_ROOT ? function() {
 33 		return{ width: document.body.clientWidth, height: document.body.clientHeight};
 34 	}:function(){
 35 		return{width:docEl.clientWidth, height:docEl.clientHeight};
 36 	});
 37 
 38     APE.mixin(APE.dom, {
 39         getScrollOffsets : getScrollOffsets,
 40         getViewportDimensions : getViewportDimensions
 41     });
 42 })();
