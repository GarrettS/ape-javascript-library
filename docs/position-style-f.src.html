 17 	/**
 18 	 * @param {HTMLElement} el you want coords of.
 19 	 * @param {HTMLElement} container to look up to.
 20 	 * @param {x:{Number}, y:{Number}} coords object to pass in.
 21 	 * @param {boolean} forceRecalc if true, forces recalculation of body scroll offsets.
 22 	 * @return {x:{Number}, y:{Number}} coords of el from container.
 23 	 * 
 24 	 * Passing in a container will improve performance in other browsers,
 25 	 * but will punish IE with a recursive call. Test accordingly.
 26 	 * 
 27 	 * Container is sometimes irrelevant. Container is irrelevant when comparing two objects' 
 28 	 * positions against one another, to see if they intersect. In this case, pass in document.
 29 	 *
 30 	 * Passing in re-used coords will greatly improve performance in all browsers.
 31 	 * There is a side effect to passing in coords:
 32 	 * If coords are passed in the viewport scroll positions are not recalculated for 
 33 	 * non-boundRect browsers.
 34 	 * For animation or drag drop operations, reuse coords.
 35 	 */
 36 	dom.getOffsetCoords = 
 37 	(document.documentElement.getBoundingClientRect) ?
 38 
 39 		// the easy way.
 40 		function(el, container, coords) {
 41 			if(!coords) 
 42 				coords = {x:0, y:0};
 43 			if(!container)
 44 				container = document;
 45 
 46 			var doc = el.ownerDocument, body = doc.body, documentElement = doc.documentElement;
 47 			
 48 			var box = el.getBoundingClientRect();
 49 			var x, y, max = Math.max;
 50 			x = box.left - max(documentElement.clientLeft, body.clientLeft)
 51 					+ max( documentElement.scrollLeft, body.scrollLeft );
 52 			y = box.top - max(documentElement.clientTop, body.clientTop)
 53 					+ max( documentElement.scrollTop, body.scrollTop );
 54 			if(container != document) {
 55 				box = getOffsetCoords(container, null);
 56 				x -= box.x;
 57 				y -= box.y;
 58 			} 
 59 			coords.x = x;
 60 			coords.y = y;
 61 
 62 			return coords;
 63 		}
 64 		:
 65 	// The shitty way, crawling up the tree.
 66 		function(el, container, coords) {
 67 
 68 			var gotCoords = true;
 69 			if(!coords) {
 70 				gotCoords = false;
 71 				coords = {x:0, y:0};
 72 			}
 73 			var doc = el.ownerDocument, body = doc.body, documentElement = doc.documentElement;
 74 
 75 			// XXX Opera bug:
 76 			// opera will inherit the offsetTop/offsetLeft body for every offsetParent.
 77 			// To amend, we set body to static, then set it back. 
 78 			// This would be done for all browsers that support currentStyle, but not IE because IE 
 79 			// supports getBoundingClientRect instead. The side effect is a style update 
 80 			// that should not be noticeable.
 81 			if(body.currentStyle) {
 82 				var isNotBodyStatic = body.currentStyle.position != "static";
 83 				if(isNotBodyStatic) {
 84 					var bs = body.style;
 85 					var oldBM = bs.margin;
 86 					var oldBL = bs.left;
 87 					var oldBT = bs.top;
 88 					bs.margin = 0;
 89 					bs.left = 0;
 90 					bs.top = 0;
 91 				}
 92 			}
 93 
 94 			// The meat of the method.
 95 			// Loop up, gathering scroll offsets on parentNodes.
 96 			// when we get to a parent that's an offsetParent, update 
 97 			// the current offsetParent marker.
 98 			if(!container) container = document;
 99 			var offsetParent = el.offsetParent, parent = el.parentNode, lastOffsetParent = offsetParent;
100 			for(var offsetLeft = el.offsetLeft, offsetTop = el.offsetTop; 
101 								parent && parent != container; parent = parent.parentNode) { 
102 				if(parent != body) {
103 					lastOffsetParent = parent;
104 					offsetLeft -= parent.scrollLeft;
105 					offsetTop -= parent.scrollTop;
106 				}
107 				if(parent == offsetParent) {
108 					offsetLeft += offsetParent.offsetLeft;
109 					offsetTop += offsetParent.offsetTop;
110 					offsetParent = parent.offsetParent; // next marker to check for offsetParent.
111 				}
112 			}
113 
114 			// XXX Opera hack, put body style back.
115 			if(isNotBodyStatic) {
116 				bs.margin = oldBM;
117 				bs.left = oldBL;
118 				bs.top = oldBT;
119 			}
120 
121 			// Workaround for bug in Mozilla and Safari and Opera whereby the 
122 			// body's margin is never included in offsetLeft/offsetTop.
123 			// This is wrong. It should be, if the body is an offsetParent.
124 			//
125 			// For opera, it's important that we do this *after* the Opera static hack.
126 			// bodyOffsetLeft is calculated once here. There is a risk that the user may dynamically 
127 			// update the body's margin or top/left values. This is unlikely.
128 			if(typeof bodyOffsetLeft == "undefined" || !gotCoords && typeof getComputedStyle == "function") {
129 				var parseInt = window.parseInt, max = Math.max;
130 				var bcs = getComputedStyle(body,'');
131 				var lastParentCS = getComputedStyle(lastOffsetParent,'');
132 				if(bcs.position == "static" && lastParentCS.position != "relative") {
133 					bodyOffsetLeft = bodyOffsetTop = 0;
134 				}
135 				else {
136 					// Safari will sometimes return "auto" for computedStyle, which results NaN.
137 					var bodyOffsetLeft = parseInt(bcs.marginLeft)||0 + parseInt(bcs.left)||0;
138 					var bodyOffsetTop = parseInt(bcs.marginTop)||0 + parseInt(bcs.top)||0;
139 
140 					bodyOffsetLeft = max(bodyOffsetLeft, body.offsetLeft);
141 					bodyOffsetTop = max(bodyOffsetTop, body.offsetTop);
142 				}
143 			}
144 			
145 			var scrollLeft = 0, scrollTop = 0, pageYOffset = window.pageYOffset;
146 			if(container == document || container == documentElement || container == body) {
147 				if( !(pageYOffset !== documentElement.scrollTop && pageYOffset === body.scrollTop) ) {
148 					var max = Math.max;
149 					scrollLeft = max( documentElement.scrollLeft||0, body.scrollLeft||0 );
150 					scrollTop = max( documentElement.scrollTop||0, body.scrollTop||0 );
151 				}
152 			}
153 
154 			coords.x = offsetLeft + bodyOffsetLeft + scrollLeft;
155 			coords.y = offsetTop + bodyOffsetTop + scrollTop;
156 
157 			return coords;
158 		}
159 
160 		// Since the body margin doesn't usually change, we can cache it here.
161 		var bodyOffsetLeft, bodyOffsetTop;

164 	/** 
165 	 * @return {boolean} true if a is vertically within b's content area (and does not overlap, top nor bottom).
166 	 */
167 	dom.isInsideElement = function(a, b) {
168 		var aTop = getOffsetCoords(a).y;
169 		var bTop = getOffsetCoords(b).y;
170 		return aTop + a.offsetHeight <= bTop + b.offsetHeight && aTop >= bTop;
171 	}
172 
173 	/** 
174 	 * @return {boolean} true if a overlaps the top of b's content area.
175 	 */
176 	dom.isAboveElement = function(a, b) {
177 		return (getOffsetCoords(a).y <= getOffsetCoords(b).y);
178 	}
179 
180 	/** 
181 	 * @return {boolean} true if a overlaps the bottom of b's content area.
182 	 */
183 	dom.isBelowElement = function(a, b) {
184 		return (getOffsetCoords(a).y + a.offsetHeight >= getOffsetCoords(b).y + b.offsetHeight);
185 	}
186 	/** findInheritedStyle tries to find a style value for the element.
187 	 * excludes: inherit|auto|transparent
188 	 * 
189 	 * @param {Element} el - element's style you want.
190 	 * @param prop {String} style property, such as backgroundColor.
191 	 * @return {String} computed style or an empty string.
192 	 */
193 	dom.findInheritedStyle = function(el, prop) {
194 		var value = "";
195 		for(var n = el; value = getStyle(n, prop); n = n.parentNode) 
196 			if(value && !noValueExp.test(value)) break;
197 		return value;
198 	};
199 	var noValueExp = /inherit|transparent/i;

201 	/** 
202 	 * Special method for a browser that supports el.filters and not style.opacity.
203 	 * @param {HTMLElement} el the element to find opacity on.
204 	 * @return {ufloat} [0-1] amount of opacity.
205 	 * calling this method on a browser that does not support filters
206 	 * results in 1 being returned.  Use dom.getStyle instead
207 	 */
208 	dom.getFilterOpacity = function(el) {
209 		var filters = el.filters;
210 		try { // Will throw error if no DXImageTransform.
211 			return filters['DXImageTransform.Microsoft.Alpha'].opacity/100
212 
213 		} catch(e) {
214 			try { // Make sure element is in the document.
215 				return filters['alpha'].opacity/100;
216 			} catch(e) {
217 				return 1;
218 			}
219 		}
220 	};

231 		if("opacity"in el.style) {
232 			el.style.opacity = i;
233 		}
234 		else if("filters"in el.style) {
235 			el.style.filter = 'alpha(opacity=' + val * 100 + ')';
236 			if(el.currentStyle && ("hasLayout"in el.currentStyle) && !el.currentStyle.hasLayout) {
237 				el.style.zoom = 1;
238 			}
239 		}

244 	 * Cross-browser adapter method for reading style. 
245 	 * Adapts for filters from opacity and styleFloat from cssFloat.
246 	 *
247 	 * @param {HTMLElement} el the element to set opacity on.
248 	 * @param {String} p the property to retrieve.
249 	 * @return {String} the computed/current style value or the empty string if no value was found.
250 	 */
251 	dom.getStyle = getStyle;
252 
253 	function getStyle(el, p) {
254 		var value = el.style[p];
255 
256 			// IE provides "medium" as default inline-style border value for all border props.
257 			// This bogus value should be ignored.
258 		if(!value || p.indexOf("border") == 0 && /thin|medium|thick/.test(value)) {
259 			if(typeof getComputedStyle == "function") {
260 				value = getComputedStyle(el, "")[p];
261 			}
262 			else if(el.currentStyle) {
263 				if(p == "opacity" && !("opacity"in el.currentStyle))
264 					value = getFilterOpacity(el);
265 				if(p == "cssFloat")
266 					p = "styleFloat";
267 				else
268 					value = el.currentStyle[p];
269 			}
270 		}
271 		return value || "";
272 	}
