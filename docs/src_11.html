1 /** 
  2  * @fileoverview dom ClassName Functions.
  3  * @namespace APE.dom
  4  * @author Garrett Smith
  5  * <p>
  6  * ClassName functions are added to APE.dom, 
  7  * Done mostly for the sake of keeping the global namespace clean.
  8  * These functions are part of dom because they are about traversing and locating elements.
  9  * </p>
 10  */
 11 
 12 /**@namespace APE.dom*/
 13 APE.namespace("APE.dom");
 14 (function() {
 15 	var dom = APE.dom;
 16 	dom.hasToken = function(s, token) { return getTokenizedExp(token,"").test(s); };
 17 	dom.removeClass = function(el, klass) {
 18 		el.className = el.className.replace(getTokenizedExp(klass, "g")," ").normalize();
 19 	};
 20 	dom.addClassConditionally = function(el, klass) {
 21 		if(!getTokenizedExp(klass).test(el.className)) el.className += " " + klass;
 22 	};
 23 	
 24 	var TokenizedExps = { };
 25 	dom.getTokenizedExp = function(token, flag){
 26 		var p = token + "$" + flag; 
 27 		return (TokenizedExps[p] || (TokenizedExps[p] = new RegExp("(^|\\s)"+token+"($|\\s)", flag)));
 28 	};
 29 
 30 	/**
 31 	 * Returns an Array of elements with the specified tagName and className.
 32 	 */
 33 	dom.getElementsWithClass = function(el, tagName, klass){
 34 		var exp = getTokenizedExp(klass,"");
 35 		var collection = el.getElementsByTagName(tagName);
 36 		var len = collection.length;
 37 		var returnedCollection = new Array(len); // initial load size.
 38 		for(var i = 0, counter = 0; i < len; i++){
 39 			if(exp.test(collection[i].className))
 40 				returnedCollection[counter++] = collection[i];
 41 		}
 42 		returnedCollection.length = counter; // trim array.
 43 		return returnedCollection;
 44 	};
 45 	
 46 	/** Finds an ancestor with specified className
 47 	 */
 48 	dom.findAncestorWithClass = function(el, klass) {
 49 		if(el == null)
 50 			return null;
 51 		var exp = getTokenizedExp(klass,"");
 52 		for(var parent = el.parentNode;parent != null;){
 53 			if( exp.test(parent.className) )
 54 				return parent;
 55 			parent = parent.parentNode;
 56 		}
 57 		return null;
 58 	};
 59 })();
