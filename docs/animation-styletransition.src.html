 45 /** 
 46  * Calculates the distance between two points.
 47  * @param {x:int, y:int} start coords
 48  * @param {x:int, y:int} end coords
 49  * @return {uint} distance from start to end.
 50  * @static
 51  */
 52 APE.anim.Animation.calculateHypoteneuse = function(start, end) {
 53 	return Math.sqrt(Math.pow(start.x - end.x, 2) + Math.pow(start.y - end.y, 2));

 56 /** 
 57  * blendValue is useful for transitioning between two numeric values.
 58  * @param v1 the first value 
 59  * @param v2 other value to mix in
 60  * @param position - ratio blend of v2:v1
 61  * @static
 62  */
 63 APE.anim.Animation.blendValue = function(v1, v2, position) {
 64 	var inverse = 1 - position;
 65 	return (v1 * inverse) + (v2 * position);
 66 };

147

149 		// desired position (pos) and the currentPosition (position).

163 			this.start();
164 			return;

167 		APE.anim.Manager.register(this);

211 	/** 
212 	 * Called by the anim.Manager

215 	_playFrame : function() {
216 
217 		var elapsed = new Date - this._startTime;
218 
219 		if(elapsed >= this.timeLimit) {
220 			this.run(this.position = this.transition(this.endOffset));
221 			this._end();
222 			return;
223 		}
224 		var distanceTraveled = (elapsed / this.duration);
225 
226 		if(this.isReversed) {
227 			distanceTraveled = -distanceTraveled;
228 		}
229 
230 		var p = this.startOffset + distanceTraveled;
231 
232 		if(this.isReversed) {
233 			p = 1 - p;
234 			this.position = 1 - this.transition(p);
235 		}
236 		else
237 			this.position = this.transition(p);
238 		if(typeof this.onplay == "function")
239 			this.onplay( this.position );
240 		this.run( this.position );
241 	},

271 /** 
272  * @class 
273  * @protected - for internal use.
274  * Manager is a Template that calls _playFrame() on each registered Animation.
275  * This object is for internal use, tightly coupled to Animation.
276  */
277 APE.anim.Manager = new function() {
278 
279 	/** 
280 	 * @private - keeps track of activeAnimations in main loop.
281 	 */
282 	var activeAnimations = [];
283 
284 	/** 
285 	 * @private - thread timers.
286 	 */
287 	var intervalId;
288 
289    /** 
290     * Registers the animation in the thread pool once.
291 	* @param {Animation} anim the animation to register.
292 	*/
293 	this.register = function(anim) {
294 		if(activeAnimations.length === 0)
295 			start.call(this);
296 		for(var i = 0; i < activeAnimations.length; i++) {
297 			if(activeAnimations[i] === anim) {
298 				return;
299 			}
300 		}
301 		activeAnimations.push(anim);
302 	};
303 	
304    /** 
305     * Registers the animation in the thread pool once.
306 	* @param {Animation} anim the animation to register.
307 	*/
308 	this.unregister = function(anim) {
309 		for(var i = 0; i < activeAnimations.length; i++) {
310 			if(activeAnimations[i] === anim) {
311 				activeAnimations.splice(i, 1);
312 			}
313 		}
314 		if(activeAnimations.length == 0) {
315 			activeAnimations = [];
316 			stop.call(this);
317 		}
318 	};
319 
320 	this.toString = function() {
321 		return"APE anim.Manager";
322 	};
323 	
324 	var startTime;
325 	/** 
326 	 * starts run();
327 	 * @private
328 	 */
329 	function start() {
330 		startTime = new Date;
331 		var delay = 17;
332 		intervalId = window.setInterval(run, delay);
333  	}
334 
335 	/** 
336 	 * Plays the frame for each animation.
337 	 * @private - starts run().
338 	 */
339 	function run() {
340 		var elapsed = new Date - startTime;
341 
342 		// Check animations.legth each iteration. Some animations 
343 		// finish before others and must be unregistered.
344 		for(var i = 0; i < activeAnimations.length; i++) {
345 		// If an error occurs, cancel the APE.anim and throw the error.
346 			var animation = activeAnimations[i];
347 			try {
348 				animation._playFrame();
349 		    } 
350 			catch(ex) {
351 				// If an error occurs, abort the anim.
352 				animation.abort(ex);
353 			}
354 		}
355 	}
356 
357 	/** 
358 	 * Called automatically when there are no more activeAnimations to run.
359 	 */
360 	function stop() {
361 		window.clearInterval(intervalId);
362 	}

390 	 * Copyright (c) Xerox Corporation, XRCE/Contextual Computing, 2002.

395 	},

503 	/**
504 	 * @method run 
505 	 * @member APE.anim.StyleTransition
506 	 * @description overrides (implements) <code>run()</code> in Animation. 
507 	 * Runs the animation, getting the correct value  
508 	 * from each ITransitionAdapter.
509 	 * This run() method gets called by Animation.
510 	 */
511 	run : function run(rationalValue) {
512 		var adapters = this.adapters;
513 		for(var i = 0, len = adapters.length; i < len; i++) {
514 			var adapter = adapters[i];

521 		if(this.inited) return;
522

524 		var adapters = [];
525 		var dom = APE.dom,
526 			TransitionAdapterFactory = APE.anim.TransitionAdapterFactory;
527 
528 		// Loop through style object to find values.
529 		for(var prop in styleObject) {

531 			if(!toValue) continue; // CSSStyleRule.

552 /**
553  * Factory for APE.anim.TransitionAdapterFactory Interface.
554  * @return {ITransitionAdapter} ITransitionAdapter for a specific type of style setting during run.
555  * The {ITransitionAdapter} implements blendTo(rationaValue).
556  * @class
557  * @private Used internally.
558  */
559 APE.anim.TransitionAdapterFactory = {
560 	// "1px", "1.1px", "-.1px" => ["-.1px", "-.1", "px"]
561 	lengthExp : /(^-?\d+|(?:-?\d?\.\d+))(px|em|ex|pt|pc|in|cm|mm|%)/i,
562 	colorExp : /color/i, 
563 	positiveLengthExp : /width|height|fontSize$/i, 
564 	filterExp : /alpha/,
565 	opacityExp : /^opacity/,
566 	intExp : /^\d+$/,
567 
568 	fromValues : function(prop, fromValue, toValue) {
569 
570 		if(this.positiveLengthExp.test( prop )) {
571 			return new this.PositiveLengthTransitionAdapter(prop, fromValue, toValue);
572 		}
573 		if(this.colorExp.test( prop )) 
574 			return new this.ColorTransitionAdapter(prop, fromValue, toValue);
575 		if(prop == 'clip')
576 			return new this.ClipTransitionAdapter(prop, fromValue, toValue);
577 		if(this.lengthExp.test( fromValue )) {
578 			return new this.LengthTransitionAdapter(prop, fromValue, toValue);
579 		}
580 		if(this.filterExp.test( prop )) 
581 			return new this.FilterTransitionAdapter(prop, fromValue, toValue);
582 		if(this.opacityExp.test( prop )) 
583 			return new this.OpacityTransitionAdapter(prop, fromValue, toValue);
584 		if(prop == "fontWeight" && this.intExp.test( fromValue ) && this.intExp.test( toValue )) {
585 			return new this.FontWeightTransitionAdapter(prop, fromValue, toValue);
586 		}
587 		// Return an object that sets toValue on completion.
588 		return new this.ThresholdTransitionAdapter(prop, fromValue, toValue);
589 	}
