  4 /**@namespace APE.dom */

  7 	var dom = APE.dom;
  8 	/** Returns true if a is completely within b's content area (and does not overlap).
  9 	 */
 10 	dom.isInsideElement  = function(a, b){
 11 		return (getOffsetTop(a) + a.offsetHeight <= getOffsetTop(b) + b.offsetHeight && 
 12 			getOffsetTop(a) >= getOffsetTop(b));
 13 	}
 14 
 15 	/** Returns true if a overlaps the top of b's content area.
 16 	 */
 17 	dom.isAboveElement = function(a, b){
 18 		return (getOffsetTop(a) <= getOffsetTop(b));
 19 	}
 20 
 21 	/** Returns true if a overlaps the bottom of b's content area.
 22 	 */
 23 	dom.isBelowElement = function(a, b){
 24 		return (getOffsetTop(a) + a.offsetHeight >= getOffsetTop(b) + b.offsetHeight);
 25 	}
 26 
 27 	// FIXME: These don't consider border offsets.
 28 	// They don't consider scroll offsets.
 29 	// Won't work in Safari <= 3 for el inside TD.
 30 	function getOffsetTop(el, container) {
 31 		if(!container) container = document;
 32 		for(var offsetTop = 0; el && el != container; el = el.offsetParent)
 33 			offsetTop += el.offsetTop;
 34 		return offsetTop;
 35 	}
 36 	// Opera 9.2:  problems in TR with position: relative
 37 	function getOffsetLeft(el, container) {
 38 		if(!container) container = document;
 39 		for(var offsetLeft = 0; el && el != container; el = el.offsetParent) 
 40 			offsetLeft += el.offsetLeft;
 41 		return offsetLeft;
 42 	}
 43 
 44 	function getOffsetCoords(el, container, coords) {
 45 		if(!coords) 
 46 			coords = {x:0, y:0};
 47 		if (el.getBoundingClientRect) { // IE
 48 			box = el.getBoundingClientRect();
 49 			var doc = el.ownerDocument, body = doc.body, documentElement = doc.documentElement;
 50 			return {x: box.left - Math.max(documentElement.clientLeft, body.clientLeft)
 51 					+ Math.max( documentElement.scrollLeft, body.scrollLeft )
 52 				, y: box.top - Math.max(documentElement.clientTop, body.clientTop)
 53 					+ Math.max( documentElement.scrollTop, body.scrollTop )};
 54 		}
 55 		else {
 56 			if(!container) container = document;
 57 			var offsetParent = null, parent = el;
 58 			for(var offsetLeft = el.offsetLeft, offsetTop = el.offsetTop; parent && parent != container; parent = parent.parentNode) {
 59 				offsetLeft += parent.scrollLeft;
 60 				offsetTop += parent.scrollTop;
 61 				if(parent == el.offsetParent) {
 62 					offsetParent = parent;
 63 					offsetLeft += offsetParent.offsetLeft;
 64 					offsetTop += offsetParent.offsetTop;
 65 					el = parent; // next marker to check for offsetParent
 66 				}
 67 			}
 68 		}
 69 		coords.x = offsetLeft;
 70 		coords.y = offsetTop;
 71 		return coords;
 72 	}
 73 
 74 	// export these in one place.
 75 	dom.getOffsetTop = getOffsetTop;
 76 	dom.getOffsetLeft = getOffsetLeft;
 77 
 78 	dom.getOffsetCoords = getOffsetCoords;

 80 	dom.getStyle = function(el, style) {
 81 		if(!el.style) return;
 82 		var p = this.toCamelCase(style);
 83 		var value = el.style[p];
 84 			// IE provides "medium" as default inline-style border value for all border props.
 85 			// This bogus value should be ignored.
 86 		if(!value || style.indexOf("border") == 0 && /thin|medium|thick/.test(value)) {
 87 			var dv = document.defaultView;
 88 			if(typeof dv == "object") // Safari 1 thinks it's "function"
 89 				value = dv.getComputedStyle(el, "").getPropertyValue(style);
 90 			else if(el.currentStyle)
 91 				value = el.currentStyle[p];
 92 		}
 93 		return value || "";
 94 	}

 96 	var noValueExp = /inherit|auto|transparent/i;
 97 
 98 	/** findInheritedStyle tries to find a style value for the element.
 99 	 * excludes: inherit|auto|transparent
100 	 * 
101 	 * @param {Element} el - element's style you want.
102 	 * @param prop {String} style property, such as backgroundColor.
103 	 * @return {String} computed style or an empty string.
104 	 */
105 	dom.findInheritedStyle = function(el, prop) {
106 		var value = "";
107 		for(var n = el; value = this.getStyle(el, prop); n = n.parentNode) 
108 			if(value && !noValueExp.test(value)) break;
109 		return value;
110 	};

112 	dom.getFilterOpacity = function(el) {
113 		var value = 100;
114 		try { // Will throw error if no DXImageTransform.
115 			value = el.filters['DXImageTransform.Microsoft.Alpha'].opacity;
116 
117 		} catch(e) {
118 			try { // Make sure element is in the document.
119 				value = el.filters('alpha').opacity;
120 			} catch(e) {
121 			}
122 		}
123 		return value/100;
124 	};

126 	// faster toCamelCase.
127 	var toCamelCaseExp = /-([a-z])/;
128 	dom.toCamelCase = function(s) {
129 		var R = RegExp;
130 		for(var exp = toCamelCaseExp; exp.test(s); s = s.replace(exp, R.$1.toUpperCase()));
131 		return s;
132 	}
