 19     var doc = this.document,
 20         documentElement = doc.documentElement,
 21         round = Math.round, max = Math.max,
 22 
 23     // Load-time constants.
 24         IS_BODY_ACTING_ROOT = documentElement && documentElement.clientWidth === 0,
 25 
 26     // IE, Safari, and Opera support clientTop. FF 2 doesn't
 27         IS_CLIENT_TOP_SUPPORTED = 'clientTop'in documentElement,
 28 
 29         TABLE = /^h/.test(documentElement.tagName) ? "table" : "TABLE",
 30 
 31         IS_CURRENT_STYLE_SUPPORTED = 'currentStyle'in documentElement,
 32 
 33     // XXX Opera <= 9.2 - parent border widths are included in offsetTop.
 34         IS_PARENT_BODY_BORDER_INCLUDED_IN_OFFSET,
 35 
 36     // XXX Opera <= 9.2 - body offsetTop is inherited to children's offsetTop
 37     // when body position is not static.
 38     // opera will inherit the offsetTop/offsetLeft of body for relative offsetParents.
 39 
 40         IS_BODY_MARGIN_INHERITED,
 41         IS_BODY_TOP_INHERITED,
 42         IS_BODY_OFFSET_EXCLUDING_MARGIN,
 43 
 44     // XXX Mozilla includes a table border in the TD's offsetLeft.
 45     // There is 1 exception: 
 46     //   When the TR has position: relative and the TD has block level content.
 47     //   In that case, the TD does not include the TABLE's border in it's offsetLeft.
 48     // We do not account for this peculiar bug. 
 49         IS_TABLE_BORDER_INCLUDED_IN_TD_OFFSET,
 50         IS_STATIC_BODY_OFFSET_PARENT_BUT_ABSOLUTE_CHILD_SUBTRACTS_BODY_BORDER_WIDTH,
 51 
 52         IS_BODY_OFFSET_IGNORED_WHEN_BODY_RELATIVE_AND_LAST_CHILD_POSITIONED,
 53 
 54         IS_CONTAINER_BODY_STATIC_INCLUDING_HTML_PADDING,
 55         IS_CONTAINER_BODY_RELATIVE_INCLUDING_HTML_PADDING_REL_CHILD,
 56         IS_CONTAINER_BODY_RELATIVE_INCLUDING_HTML_PADDING_ABS_CHILD,
 57         IS_CONTAINER_BODY_INCLUDING_HTML_MARGIN,
 58 
 59         // In Safari 2.0.4, BODY can have offsetTop when offsetParent is null.
 60         // but offsetParent will be HTML (root) when HTML has position. 
 61         // IS_BODY_OFFSET_TOP_NO_OFFSETPARENT,
 62         
 63         IS_COMPUTED_STYLE_SUPPORTED = doc.defaultView 
 64             && typeof doc.defaultView.getComputedStyle != "undefined",
 65         getBoundingClientRect = "getBoundingClientRect",
 66         relative = "relative",
 67         borderTopWidth = "borderTopWidth",
 68         borderLeftWidth = "borderLeftWidth",
 69         positionedExp = /^(?:r|a)/,
 70         absoluteExp = /^(?:a|f)/;

 72     /**
 73      * @memberOf APE.dom
 74      * @param {HTMLElement} el you want coords of.
 75      * @param {HTMLElement} positionedContainer container to look up to. The container must have 
 76      * position: (relative|absolute|fixed);
 77      *
 78      * @param {x:Number, y:Number} coords object to pass in.
 79      * @return {x:Number, y:Number} coords of el from container.
 80      * 
 81      * Passing in a container will improve performance in browsers that don't support 
 82      * getBoundingClientRect, but those that do will have a recursive call. Test accordingly.
 83      * <p>
 84      * Container is sometimes irrelevant. Container is irrelevant when comparing positions 
 85      * of objects who do not share a common ancestor. In this case, pass in document.
 86      * </p>
 87      *<p>
 88      * Passing in re-used coords can improve performance in all browsers.
 89      * There is a side effect to passing in coords:
 90      * For drag drop operations, reuse coords:
 91      *</p>
 92      * <pre>
 93      * // Update our coords:
 94      * dom.getOffsetCoords(el, container, this.coords);
 95      * </pre>
 96      * Where <code>this.coords = {};</code>
 97      */
 98     function getOffsetCoords(el, container, coords) {
 99 
100         var doc = el.ownerDocument,
101             documentElement = doc.documentElement,
102             body = doc.body;
103 
104         if(!container)
105             container = doc;
106 
107         if(!coords) 
108             coords = {x:0, y:0};
109 
110         if(el === container) {
111             coords.x = coords.y = 0;
112             return coords;
113         }
114         if(el[getBoundingClientRect]) {
115             
116             // In BackCompat mode, body's border goes to the window. BODY is ICB.
117             var rootBorderEl = IS_BODY_ACTING_ROOT ? body : documentElement,
118                 box = el[getBoundingClientRect](),
119                 x = box.left + max( documentElement.scrollLeft, body.scrollLeft ),
120                 y = box.top + max( documentElement.scrollTop, body.scrollTop ),
121                 bodyCurrentStyle,
122                 borderTop = rootBorderEl.clientTop,
123                 borderLeft = rootBorderEl.clientLeft;
124 
125             if(IS_CLIENT_TOP_SUPPORTED) {
126                 x -= borderLeft;
127                 y -= borderTop;
128             }
129             if(container !== doc) {
130                 box = getOffsetCoords(container, null);
131                 x -= box.x;
132                 y -= box.y;
133                 if(IS_BODY_ACTING_ROOT && container === body && IS_CLIENT_TOP_SUPPORTED) {
134                     x -= borderLeft;
135                     y -= borderTop;
136                 }
137             }
138 
139             if(IS_BODY_ACTING_ROOT && IS_CURRENT_STYLE_SUPPORTED 
140                 && container != doc && container !== body) {
141                 bodyCurrentStyle = body.currentStyle;
142                 x += parseFloat(bodyCurrentStyle.marginLeft)||0 + 
143                     parseFloat(bodyCurrentStyle.left)||0;
144                 y += parseFloat(bodyCurrentStyle.marginTop)||0 +
145                      parseFloat(bodyCurrentStyle.top)||0;
146             }
147             coords.x = x;
148             coords.y = y;
149 
150             return coords;
151         }
152 
153     // Crawling up the tree. 
154         else if(IS_COMPUTED_STYLE_SUPPORTED){
155             
156             var offsetLeft = el.offsetLeft, 
157                 offsetTop = el.offsetTop,
158                 defaultView = doc.defaultView,
159                 cs = defaultView.getComputedStyle(el, '');
160             if(cs.position == "fixed") {
161                 coords.x = offsetLeft + documentElement.scrollLeft;
162                 coords.y = offsetTop + documentElement.scrollTop;
163                 return coords;
164             }
165             var bcs = defaultView.getComputedStyle(body,''),
166                 isBodyStatic = !positionedExp.test(bcs.position),
167                 lastOffsetParent = el, 
168                 parent = el.parentNode,
169                 offsetParent = el.offsetParent;
170             
171         // Main loop -----------------------------------------------------------------------
172             // Loop up, gathering scroll offsets on parentNodes.
173             // when we get to a parent that's an offsetParent, update 
174             // the current offsetParent marker.
175             for( ; parent && parent !== container; parent = parent.parentNode) { 
176                 if(parent !== body && parent !== documentElement) {
177                     offsetLeft -= parent.scrollLeft;
178                     offsetTop -= parent.scrollTop;
179                 }
180                 if(parent === offsetParent) {
181                     // If we get to BODY and have static position, skip it.
182                     if(parent === body && isBodyStatic); 
183                     else {
184 
185                         // XXX Mozilla; Exclude static body; if static, it's offsetTop will be wrong.
186                         // Include parent border widths. This matches behavior of clientRect approach.
187                         // XXX Opera <= 9.2 includes parent border widths. 
188                         // See IS_PARENT_BODY_BORDER_INCLUDED_IN_OFFSET below. 
189                         if( !IS_PARENT_BODY_BORDER_INCLUDED_IN_OFFSET && 
190                             ! (parent.tagName === TABLE && IS_TABLE_BORDER_INCLUDED_IN_TD_OFFSET)) {
191                                 var pcs = defaultView.getComputedStyle(parent, "");
192                                 // Mozilla doesn't support clientTop. Add borderWidth to the sum. 
193                                 offsetLeft += parseFloat(pcs[borderLeftWidth])||0;
194                                 offsetTop += parseFloat(pcs[borderTopWidth])||0;
195                         }
196                         if(parent !== body) {
197                             offsetLeft += offsetParent.offsetLeft;
198                             offsetTop += offsetParent.offsetTop;
199                             lastOffsetParent = offsetParent;
200                             offsetParent = parent.offsetParent; // next marker to check for offsetParent.
201                         }
202                     }
203                 }
204             }
205         
206             //--------Post - loop, body adjustments----------------------------------------------
207             // Complications due to CSSOM Views - the browsers try to implement a contradictory
208             // spec: http://www.w3.org/TR/cssom-view/#offset-attributes
209 
210             // XXX Mozilla, Safari 3, Opera: body margin is never
211             // included in body offsetLeft/offsetTop.
212             // This is wrong. Body's offsetTop should work like any other element.
213             // In Safari 2.0.4, BODY can have offsetParent, and even 
214             // if it doesn't, it can still have offsetTop.
215             // But Safari 2.0.4 doubles offsetTop for relatively positioned elements 
216             // and this script does not account for that.
217 
218             // XXX Mozilla: When body has a border, body's offsetTop === negative borderWidth;
219             // Don't use body.offsetTop.
220             var bodyOffsetLeft = 0,
221                 bodyOffsetTop = 0,
222                 isLastElementAbsolute,
223                 isLastOffsetElementPositioned,
224                 isContainerDocOrDocEl = container === doc || container === documentElement,
225                 dcs,
226                 lastOffsetPosition;
227 
228             // If the lastOffsetParent is document, 
229             // it is not positioned (and hence, not absolute).
230             if(lastOffsetParent != doc) {
231                 lastOffsetPosition = defaultView.getComputedStyle(lastOffsetParent,'').position;
232                 isLastElementAbsolute = absoluteExp.test(lastOffsetPosition);
233                 isLastOffsetElementPositioned = isLastElementAbsolute || 
234                     positionedExp.test(lastOffsetPosition);
235             }
236 
237             // do we need to add margin? 
238             if( 
239                 (lastOffsetParent === el && el.offsetParent === body && !IS_BODY_MARGIN_INHERITED
240                 && container !== body && !(isBodyStatic && IS_BODY_OFFSET_EXCLUDING_MARGIN))
241                 || (IS_BODY_MARGIN_INHERITED && lastOffsetParent === el && !isLastOffsetElementPositioned)
242                 || !isBodyStatic 
243                 && isLastOffsetElementPositioned
244                 && IS_BODY_OFFSET_IGNORED_WHEN_BODY_RELATIVE_AND_LAST_CHILD_POSITIONED
245                 && isContainerDocOrDocEl) {                    
246                     bodyOffsetTop += parseFloat(bcs.marginTop)||0;
247                     bodyOffsetLeft += parseFloat(bcs.marginLeft)||0;
248             }
249 
250             // Case for padding on documentElement.
251             if(container === body) {
252                 dcs = defaultView.getComputedStyle(documentElement,'');
253                 if( 
254                     (!isBodyStatic && 
255                         ((IS_CONTAINER_BODY_RELATIVE_INCLUDING_HTML_PADDING_REL_CHILD && !isLastElementAbsolute)
256                         || 
257                         (IS_CONTAINER_BODY_RELATIVE_INCLUDING_HTML_PADDING_ABS_CHILD && isLastElementAbsolute))
258                     )
259                     || isBodyStatic && IS_CONTAINER_BODY_STATIC_INCLUDING_HTML_PADDING
260                     ) {
261                     bodyOffsetTop -= parseFloat(dcs.paddingTop)||0;
262                     bodyOffsetLeft -= parseFloat(dcs.paddingLeft)||0;
263                 }
264 
265                 if(IS_CONTAINER_BODY_INCLUDING_HTML_MARGIN){
266                     if(!isLastOffsetElementPositioned 
267                         || isLastOffsetElementPositioned && !isBodyStatic)
268                     bodyOffsetTop -= parseFloat(dcs.marginTop)||0;
269                     bodyOffsetLeft -= parseFloat(dcs.marginLeft)||0;
270                 }
271             }
272             if(isBodyStatic) { 
273                 // XXX Safari subtracts border width of body from element's offsetTop (opera does it, too)
274                 if(IS_STATIC_BODY_OFFSET_PARENT_BUT_ABSOLUTE_CHILD_SUBTRACTS_BODY_BORDER_WIDTH
275                 // XXX: Safari will use HTML for containing block (CSS),
276                 // but will subtract the body's border from the body's absolutely positioned 
277                 // child.offsetTop. Safari reports the child's offsetParent is BODY, but 
278                 // doesn't treat it that way (Safari bug).
279                     || (!isLastElementAbsolute && !IS_PARENT_BODY_BORDER_INCLUDED_IN_OFFSET
280                         && isContainerDocOrDocEl)) {
281                            bodyOffsetTop += parseFloat(bcs[borderTopWidth]);
282                            bodyOffsetLeft += parseFloat(bcs[borderLeftWidth]);
283                 }
284             }
285 
286             // body is positioned, and if it excludes margin,
287             // it's probably partly using the AVK-CSSOM disaster.
288             else if(IS_BODY_OFFSET_EXCLUDING_MARGIN) {
289                 if(isContainerDocOrDocEl) {
290                     if(!IS_BODY_TOP_INHERITED) {
291 
292                         // If the body is positioned, add its left and top value.
293                          bodyOffsetTop += parseFloat(bcs.top)||0;
294                          bodyOffsetLeft += parseFloat(bcs.left)||0;
295                         
296                         // XXX: Opera normally include the parentBorder in offsetTop.
297                         // We have a preventative measure in the loop above.
298                         if(isLastElementAbsolute && IS_PARENT_BODY_BORDER_INCLUDED_IN_OFFSET) {
299                             bodyOffsetTop += parseFloat(bcs[borderTopWidth]);
300                             bodyOffsetLeft += parseFloat(bcs[borderLeftWidth]);
301                         }
302                     }
303 
304                     // Padding on documentElement is not included,
305                     // but in this case, we're searching to documentElement, so we 
306                     // have to add it back in.
307                     if(container === doc && !isBodyStatic 
308                         && !IS_CONTAINER_BODY_RELATIVE_INCLUDING_HTML_PADDING_REL_CHILD) {
309                         if(!dcs) dcs = defaultView.getComputedStyle(documentElement,'');
310                         bodyOffsetTop += parseFloat(dcs.paddingTop)||0;
311                         bodyOffsetLeft += parseFloat(dcs.paddingLeft)||0;
312                     }
313                 }
314                 else if(IS_BODY_TOP_INHERITED) {
315                     bodyOffsetTop -= parseFloat(bcs.top);
316                     bodyOffsetLeft -= parseFloat(bcs.left);
317                 }
318                 if(IS_BODY_MARGIN_INHERITED && (!isLastOffsetElementPositioned || container === body)) {
319                     bodyOffsetTop -= parseFloat(bcs.marginTop)||0;
320                     bodyOffsetLeft -= parseFloat(bcs.marginLeft)||0;
321                 }
322             }
323             coords.x = round(offsetLeft + bodyOffsetLeft);
324             coords.y = round(offsetTop + bodyOffsetTop);
325 
326             return coords;
327         }
328     }

332     (function(){
333     var waitForBodyTimer = setInterval(function() {
334         var body = doc.body;
335         if(!body) return;
336         clearInterval(waitForBodyTimer);
337         var marginTop = "marginTop", position = "position", padding = "padding", 
338             stat = "static",
339             border = "border", s = body.style, 
340             bCssText = s.cssText,
341             bv = '1px solid transparent',
342             z = "0",
343             one = "1px",
344             offsetTop = "offsetTop",
345             ds = documentElement.style,
346             dCssText = ds.cssText,
347             x = doc.createElement('div'),
348             xs = x.style,
349             table = doc.createElement(TABLE);
350 
351         s[padding] = s[marginTop] = s.top = z;
352         ds.position = stat;
353 
354         s[border] = bv;
355 
356         xs.margin = z;
357         xs[position] = stat;
358 
359         // insertBefore - to avoid environment conditions with bottom script
360         // where appendChild would fail.
361         x = body.insertBefore(x, body.firstChild);
362         IS_PARENT_BODY_BORDER_INCLUDED_IN_OFFSET = (x[offsetTop] === 1);
363         
364         s[border] = z;
365 
366         // Table test.
367         table.innerHTML = "<tbody><tr><td>x</td></tr></tbody>";
368         table.style[border] = "7px solid";
369         table.cellSpacing = table.cellPadding = z;
370 
371         body.insertBefore(table, body.firstChild);
372         IS_TABLE_BORDER_INCLUDED_IN_TD_OFFSET = table.getElementsByTagName("td")[0].offsetLeft === 7;
373 
374         body.removeChild(table);
375 
376         // Now add margin to determine if body offsetTop is inherited.
377         s[marginTop] = one;
378         s[position] = relative;
379         IS_BODY_MARGIN_INHERITED = (x[offsetTop] === 1);
380         
381         //IS_BODY_OFFSET_TOP_NO_OFFSETPARENT = body.offsetTop && !body.offsetParent;
382 
383         IS_BODY_OFFSET_EXCLUDING_MARGIN = body[offsetTop] === 0;
384         s[marginTop] = z;
385         s.top = one;
386         IS_BODY_TOP_INHERITED = x[offsetTop] === 1;
387 
388         s.top = z;
389         s[marginTop] = one;
390         s[position] = xs[position] = relative;
391         IS_BODY_OFFSET_IGNORED_WHEN_BODY_RELATIVE_AND_LAST_CHILD_POSITIONED = x[offsetTop] === 0;
392 
393         xs[position] = "absolute";
394         s[position] = stat;
395         if(x.offsetParent === body) {
396             s[border] = bv;
397             xs.top = "2px";
398             // XXX Safari gets offsetParent wrong (says 'body' when body is static,
399             // but then positions element from ICB and then subtracts body's clientWidth.
400             // Safari is half wrong.
401             //
402             // XXX Mozilla says body is offsetParent but does NOT subtract EL's offsetLeft/Top.
403             // Mozilla is completely wrong.
404             IS_STATIC_BODY_OFFSET_PARENT_BUT_ABSOLUTE_CHILD_SUBTRACTS_BODY_BORDER_WIDTH = x[offsetTop] === 1;
405             s[border] = z;
406             
407             xs[position] = relative;
408             ds[padding] = one;
409             s[marginTop] = z;
410 
411             IS_CONTAINER_BODY_STATIC_INCLUDING_HTML_PADDING = x[offsetTop] === 3;
412 
413             // Opera does not respect position: relative on BODY.
414             s[position] = relative;
415             IS_CONTAINER_BODY_RELATIVE_INCLUDING_HTML_PADDING_REL_CHILD = x[offsetTop] === 3;
416 
417             xs[position] = "absolute";
418             IS_CONTAINER_BODY_RELATIVE_INCLUDING_HTML_PADDING_ABS_CHILD = x[offsetTop] === 3;
419             
420             ds[padding] = z;
421             ds[marginTop] = one;
422 
423             // Opera inherits HTML margin when body is relative and child is relative or absolute.
424             IS_CONTAINER_BODY_INCLUDING_HTML_MARGIN = x[offsetTop] === 3;
425         }
426 
427         // xs.position = "fixed";
428         // FIXED_HAS_OFFSETPARENT = x.offsetParent != null;
429 
430         body.removeChild(x);
431         s.cssText = bCssText||"";
432         ds.cssText = dCssText||"";

437   // TODO: add an optional commonAncestor parameter to the below.
438     /** 
439      * @memberOf APE.dom
440      * @return {boolean} true if a is vertically within b's content area (and does not overlap, 
441      * top nor bottom).
442      */
443     function isInsideElement(a, b) {
444         var aTop = getOffsetCoords(a).y,
445             bTop = getOffsetCoords(b).y;
446         return aTop + a.offsetHeight <= bTop + b.offsetHeight && aTop >= bTop;
447     }
448 
449     /** 
450      * @memberOf APE.dom
451      * @return {boolean} true if a overlaps the top of b's content area.
452      */
453     function isAboveElement(a, b) {
454         return (getOffsetCoords(a).y <= getOffsetCoords(b).y);
455     }
456 
457     /** 
458      * @memberOf APE.dom
459      * @return {boolean} true if a overlaps the bottom of b's content area.
460      */
461     function isBelowElement(a, b) {
462         return (getOffsetCoords(a).y + a.offsetHeight >= getOffsetCoords(b).y + b.offsetHeight);
463     }
464     
465     // release from closure.
466     isInsideElement = isAboveElement = isBelowElement = null;
