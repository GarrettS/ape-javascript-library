  7 APE.namespace("APE.resize");
  8 
  9 APE.resize.Resizable = function(id) {
 10     this.id = (typeof id == "string" ? id : id.id);
 11     this.proxyId = this.id + "-proxy";
 12     this.create();
 13 
 14     APE.MouseTracker.register(this, this.proxyId);
 15 };
 16 
 17 APE.resize.Resizable.getById = APE.getById;
 18 APE.resize.Resizable.getByNode = APE.getByNode;
 19 
 20 APE.resize.Resizable.prototype = {
 21 
 22     create : function() {
 23         var el = document.getElementById(this.id),
 24             APE = window.APE,
 25             dom = APE.dom,
 26             position = dom.getStyle(el, "position"),
 27             proxy = document.createElement("div");
 28 
 29         proxy.id = this.proxyId;
 30         APE.EventPublisher.add(proxy, "onmouseover", this.mouseOver, this);
 31         proxy.className = "ape-resize-container";
 32 
 33         if(!/^(?:r|a|f)/.test(position))
 34             position = "relative";
 35         proxy.style.position = el.style.position = position;
 36 
 37         
 38         if(el.currentStyle && "hasLayout"in el.currentStyle &&
 39             !el.currentStyle.hasLayout) {
 40             el.style.zoom = 1;
 41         }
 42 
 43         el.parentNode.insertBefore(proxy, el);
 44         proxy.appendChild(el);
 45         this.pack();
 46     },
 47     
 48     pack : function() {
 49         var proxy = document.getElementById(this.proxyId),
 50             ps = proxy.style,
 51             el = document.getElementById(this.id);
 52         ps.width = el.offsetWidth;
 53         ps.height = el.offsetHeight;
 54     },
 55 
 56     getResizeType : function(e, proxyEl) {
 57         var resizeType = "",
 58             clientHeight = proxyEl.clientHeight,
 59             clientWidth = proxyEl.clientWidth;
 60 
 61         if(ex < proxyEl.clientTop) {
 62             resizeType = "n";
 63         }
 64         else if(ex > clientHeight) {
 65             resizeType = "s";
 66         }
 67 
 68         if(e.clientX > clientWidth) {
 69             resizeType += 'e';
 70         }
 71         else if(e.clientX < proxyEl.clientLeft) {
 72             resizeType += 'w';
 73         }
 74         return resizeType;
 75     },
 76 
 77     mouseDown : function(e) {
 78         var target = APE.dom.Event.getTarget(e),
 79             proxyEl = document.getElementById(this.id);
 80         this.getResizeType(e, proxyEl);
 81         console.log(e);
 82         if(target.id === this.proxyId) {
 83             return true;
 84         }
 85     },
 86 
 87     isInHandle : function(target) {
 88         return target.id === this.proxyId;
 89     },
 90 
 91     mouseOver : function(e) {
 92         var target = this.getEventTarget(e),
 93             resizeType;
 94         if(this.isInHandle(target)) {
 95             resizeType = this.getResizeType(e, document.getElementById(this.proxyId));
 96         console.log(resizeType);
 97             target.style.cursor = resizeType + "-resize";
 98         }
 99         else target.style.cursor = "";
100     },
101 
102     mouseMove : function(e) {
103         var proxy = document.getElementById(this.proxyId),
104             ps = proxy.style,
105             el = document.getElementById(this.id);
106 
107         console.log(e);
108     },
109 
110     mouseUp : function(e) {
111         console.log(e);
112         return true;
113     },
114 
115     getEventTarget : APE.dom.Event.getTarget
116 };
