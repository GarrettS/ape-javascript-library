 29     /** findInheritedStyle tries to find a cascaded style value for the element.
 30      * If the value is inherit|transparent, it looks up the tree, recursively.
 31      * @memberOf APE.dom
 32      * 
 33      * @param {Element} el - element's style you want.
 34      * @param prop {String} style property, such as backgroundColor.
 35      * @param {String} [units] optional unit to search for. Example: "em".
 36      * @return {String} computed style or an empty string.
 37      */
 38     function findInheritedStyle(el, prop, units) {
 39         var value = "", n = el;
 40         for( ; value = getCascadedStyle(n, prop, units); n = n.parentNode) 
 41             if(value && !noValueExp.test(value)) break;
 42         return value;
 43     }

 47     /** 
 48      * Special method for a browser that supports el.filters and not style.opacity.
 49      * @memberOf APE.dom
 50      * @param {HTMLElement} el the element to find opacity on.
 51      * @return {ufloat} [0-1] amount of opacity.
 52      * calling this method on a browser that does not support filters
 53      * results in 1 being returned.  Use dom.getStyle or dom.getCascadedStyle instead
 54      */
 55      function getFilterOpacity(el) {
 56         var filters = el.filters;
 57         if(!filters) return"";
 58         try { // Will throw error if no DXImageTransform.
 59             return filters['DXImageTransform.Microsoft.Alpha'].opacity/100;
 60 
 61         } catch(e) {
 62             try { 
 63                 return filters('alpha').opacity/100;
 64             } catch(e) {
 65                 return 1;
 66             }
 67         }
 68     }

 71      * Cross-browser adapter method for style.filters vs style.opacity.

 74      * @param {ufloat} i [0-1] the amount of opacity.

 77      function setOpacity(el, i) {
 78         var s = el[style], cs;
 79         if("opacity"in s) {
 80             s.opacity = i;
 81         }
 82         else if("filter"in s) {
 83             cs = el[currentStyle];
 84             s.filter = 'alpha(opacity=' + (i * 100) + ')';
 85             if(cs && ("hasLayout"in cs) && !cs.hasLayout) {
 86                 style.zoom = 1;
 87             }
 88         }

 91     /** 
 92      * @memberOf APE.dom
 93      * @name getStyle
 94      * 
 95      * @function
 96      * @description returns the computed style of property <code>p</code> of <code>el</code>.
 97      * Returns different results in IE, so user beware! If your 
 98      * styleSheet has units like "em" or "in", this method does 
 99      * not attempt to convert those to px.
100      *
101      * Use "cssFloat" for getting an element's float and special 
102      * "filters" treatment for "opacity".
103      * 
104      * @param {HTMLElement} el the element to set opacity on.
105      * @param {String} p the property to retrieve.
106      * @return {String} the computed style value or the empty string if no value was found.
107      */
108     function _getComputedStyle(el, p) {
109         var value = "", cs, matches, splitVal, i, len, doc = el.ownerDocument, 
110             defaultView = doc.defaultView;
111         if(IS_COMPUTED_STYLE_SUPPORTED) {
112             cs = defaultView[getCS](el, "");
113             if(p == "borderRadius" && !("borderRadius"in cs)) {
114                 p = "MozBorderRadius"in cs ? "MozBorderRadius" : 
115                     "WebkitBorderRadius"in cs ? "WebkitBorderRadius" : "";
116             }
117 
118             if(!(p in cs)) return "";
119             value = cs[p];
120             if(value === "") {
121                 // would try to get a rect, but Webkit doesn't support that.
122                 value = (tryGetShorthandValues(cs, p)).join(" ");
123             }
124         }
125         else if(currentStyle in el) {
126             cs = el[currentStyle];
127             if(p == "opacity" && !("opacity"in el[currentStyle]))
128                 value = getFilterOpacity(el);
129             else {
130                 if(p == "cssFloat")
131                     p = "styleFloat";
132                 value = cs[p];
133 
134                 if(p == "clip" && !value && ("clipTop"in cs)) {
135                     value = getCurrentStyleClipValues(el, cs);
136                 }
137                 else if(value == "auto") 
138                     value = getCurrentStyleValueFromAuto(el, p);
139                 else if(!(p in cs)) return "";
140             }
141             matches = nonPixelExp.exec(value);
142             if(matches) {
143                 splitVal = value.split(" ");
144                 splitVal[0] = convertNonPixelToPixel( el, matches);
145                 for(i = 1, len = splitVal.length; i < len; i++) {
146                     matches = nonPixelExp.exec(splitVal[i]);
147                     splitVal[i] = convertNonPixelToPixel( el, matches);
148                 }
149                 value = splitVal.join(" ");
150             }
151         }
152         return value;
153     }

156         var values = [], i = 0, prop;
157         for( ;i < 4; i++){
158             prop = props[i];
159             clipValue = cs['clip'+prop];
160             if(clipValue == "auto") {
161                 clipValue = (prop == "Left" || prop == "Top" ? "0px" : prop == "Right" ? 
162                     el.offsetWidth + px : el.offsetHeight + px);
163             }
164             values.push(clipValue);
165         }
166         return {
167             top:values[0], right:values[1], bottom:values[2], left:values[3],
168             toString : function() {return 'rect(' + values.join(' ')+')';}
169         };

177     /** 
178      * @memberOf APE.dom
179      *
180      * @description Cross-browser adapter method for reading style. 
181      * Adapts for filters from opacity and styleFloat from cssFloat.
182      * For browsers that support computed styles, but not <code>currentStyle</code>
183      * (Firefox/Safari) pass in a <code>desiredUnit</code> of either <code>em<code> or 
184      * <code>ex</code>
185      * @memberOf APE.dom
186      *
187      * Tries to get a style of the attribute el[p]. If not found, uses 
188      * getComputedStyle or currentStyle. getComputedStyle returns a pixel unit, but 
189      * we have a patch to do the math.
190      * 
191      * <p>
192      * Performance tip: performance-critical operations can reference the 
193      * element's style first: <code>var val = el.style[p] || getComputedStyle(el);</code>
194      * </p>
195      * 
196      * @param {HTMLElement} el the element to set opacity on.
197      * @param {String} p the property to retrieve.
198      * @param {String} desiredUnit one of: ["em", ] the type of unit to retrieve/compute - 
199      * this is not really supported in IE - it is mostly for browsers that don't support 
200      * currentStyle.
201      * @return {String} the cascaded style value or the empty string if no value was found.
202      */
203     function getCascadedStyle(el, p, desiredUnit) {
204         
205         var s = el[style],
206             value = s[p]||"";
207         if(value && multiLengthPropExp.test(p)) {
208             value = normalizeValue(value);
209         }
210 
211         // IE provides "medium" as default inline-style border value for all border props.
212         // This bogus value should be ignored.
213 
214         if(!value || (desiredUnit && value.indexOf(desiredUnit) === -1) 
215             || p.indexOf("border") === 0 && borderWidthExp.test(value)) {
216 
217             if(currentStyle in el) {
218                 value = getCascadedFromCurrent(el, p, desiredUnit);
219             }
220 
221             else {
222                 if(borderRadiusExp.test(p))
223                     p = borderRadiusExp.exec(p)[0];
224                 value = getCascadedFromComputed(el, p, desiredUnit);
225             }
226         }
227         return value;
228     }

230     function getCascadedFromCurrent(el, p, desiredUnit) {
231 
232         var curSty = el[currentStyle], value = "", unitAdapter, doc = el.ownerDocument, 
233             defaultView = doc.defaultView;
234         
235         if(p == "opacity") {
236             
237             // currentStyle is pretty fucked in Opera.
238             // returns "1". So go for getComputedStyle first.
239             if(IS_COMPUTED_STYLE_SUPPORTED)
240                 value = defaultView[getCS](el,'').opacity;
241             else if(!("opacity"in curSty)) {
242                 value = getFilterOpacity(el);
243             }
244         }
245         else if(p == 'clip' && !curSty[p] && 'clipTop'in curSty) {
246             value = getCurrentStyleClipValues(el, curSty);
247         }
248         else {
249 
250             // We've tried clip and opacity now, so it seems that the property 
251             // does not exist, ala "WebkitBorderRadius" in IE.
252             if(!(p in curSty)) return"";
253             if(floatExp.test(p))
254                 p = floatProp;
255             value = el[style][p] || curSty[p];
256 
257             if(value == "auto") 
258                 value = getCurrentStyleValueFromAuto(el, p) || value;
259 
260         }
261 
262         if(desiredUnit && value.indexOf(desiredUnit) == -1) {
263             // Opera 9.2 royally fucked up currentStyle.
264             // calls floor() on some values
265             // If we ended up here, we have something like "0em",
266             // so we pretend to have "0px" and then use a UnitAdapter.
267             if(isCurrentStyleFloored && unitExp.test(value) && IS_COMPUTED_STYLE_SUPPORTED) {
268                 var cs = defaultView[getCS](el, p);
269                 unitAdapter = getAdapterFor(el, p, desiredUnit);
270                 if(unitAdapter) {
271                     value = unitAdapter.fromPx(el.parentNode, p, cs[p], cs);
272                 }
273             }
274             else if(value == 0 && desiredUnit)
275                 value = "0" + desiredUnit;
276             else {
277                 unitAdapter = getAdapterFor(el, p, desiredUnit);
278                 value = unitAdapter.fromPx(el.parentNode, p);
279             }
280 
281         }
282         return value;
283     }

285     function getCurrentStyleValueFromAuto(el, p) {
286         
287         var s = el[style], v, borderWidth, doc = el.ownerDocument;
288         if("pixelWidth"in s && pixelDimensionExp.test(p)) {
289             var pp = "pixel" + (p.charAt(0).toUpperCase()) + p.substring(1);
290             v = s[pp];
291             if(v === 0) {
292                 if(p == "width") {
293                     borderWidth = parseFloat(_getComputedStyle(el, "borderRightWidth"))||0;
294                     paddingWidth = parseFloat(_getComputedStyle(el, "paddingLeft"))||0
295                         + parseFloat(_getComputedStyle(el, "paddingRight"))||0;
296 
297                     return el.offsetWidth - el.clientLeft - borderWidth - paddingWidth + "px";
298                 } 
299                 else if(p == "height") {
300                     borderWidth = parseFloat(_getComputedStyle(el, "borderBottomWidth"))||0;
301                     paddingWidth = parseFloat(_getComputedStyle(el, "paddingTop"))||0
302                         + parseFloat(_getComputedStyle(el, "paddingBottom"))||0;
303                     return el.offsetHeight - el.clientTop - borderWidth + "px";
304                 }
305             }
306             return s[pp] + "px";
307         }
308         if(p == "margin" && el[currentStyle].position != "absolute" && 
309           doc.compatMode != "BackCompat") {
310             v = parseFloat(_getComputedStyle(el.parentNode, 'width')) - el.offsetWidth;
311             if(v == 0) return "0px";
312             v = "0px " + v;
313             return v + " " + v;
314         }
315         
316         // Can't get borderWidth because we only have clientTop and clientLeft.
317     }

319     function getCascadedFromComputed(el, p, desiredUnit) {
320 
321         if(IS_COMPUTED_STYLE_SUPPORTED) {
322             var defaultView = el.ownerDocument.defaultView,
323                 cs = defaultView[getCS](el,''),
324                 value = cs[p],
325                 valueSplit,
326                 i = 0,
327                 len,
328                 parentNode,
329                 unitAdapter,
330                 valuei;
331 
332             // Always return a string. Even for bogus properties.
333             if(!(p in cs)) return "";
334 
335             if(value === "") {
336                 valueSplit = tryGetShorthandValues(cs, p);
337             }
338             else if(parseFloat(value) == 0 && desiredUnit) {
339                 return "0" + desiredUnit;
340             }
341 
342             // The desiredUnit won't match computedStyle, but might match the 
343             // element's HTML style Attribute.
344             if(desiredUnit) { 
345                 if(!valueSplit)
346                     valueSplit = [value];
347                 unitAdapter = getAdapterFor(el, p, desiredUnit);
348 
349                 if(unitAdapter) {
350                     parentNode = el.parentNode;
351                     for(len = valueSplit.length; i < len; i++) {
352                         valuei = valueSplit[i];
353                         if(valuei == "0") valueSplit[i] = "0" + desiredUnit;
354                         else if(!unitAdapter.exp.test(valuei))
355                             valueSplit[i] = unitAdapter.fromPx(parentNode, p, valuei, cs);
356                     }
357                 }
358             }
359             if(valueSplit)
360                 value = normalizeValue(valueSplit.join(" "));
361             return value;
362         }
363     }

365     /** 
366      * takes a "1px 1px 1px 1px" string and tries to reduce to 
367      * 1px. This makes calculating some values easier.
368      */
369     function normalizeValue(value) {
370         var values = value.split(" "),
371             i = 1,
372             len,
373             allEqual = true,
374             p0 = values[0];
375         for(len = values.length-1; i < len; i++) {
376             if(!allEqual) break;
377             allEqual = values[i] == values[i+1];
378         }
379         if(allEqual) 
380             value = p0;
381         return value;
382     }

384     /** 
385      * Tries to get a shorthand value for margin|padding|borderWidth. 
386      * @return  {[string]} Either 4 values or, if all four values are equal,
387      * then one collapsed value (in an array).
388      */
389     function tryGetShorthandValues(cs, p) {
390         var multiMatch = multiLengthPropExp.exec(p),
391             prefix, suffix, 
392             prevValue, nextValue, 
393             values,
394             allEqual = true, 
395             propertyList,
396             i = 1;
397         
398         if(multiMatch && multiMatch[0]) {
399             propertyList = props;
400             prefix = multiMatch[1]||multiMatch[0];
401             suffix = multiMatch[2] || ""; // ["borderWidth", "border", "Width"]
402         }
403         else if(borderRadiusExp.test(p)) {
404            propertyList = cornerProps;
405             prefix = borderRadiusExp.exec(p)[0];
406             suffix = ""; 
407         }
408         else return [""];
409 
410         prevValue = cs[prefix + propertyList[0] + suffix ];
411         values = [prevValue];
412 
413         while(i < 4) {
414             nextValue = cs[prefix + propertyList[i] + suffix];
415             allEqual = allEqual && nextValue == prevValue;
416             prevValue = nextValue;
417             values[i++] = nextValue;
418         }
419         if(allEqual)
420             return [prevValue];
421         return values;
422     }

448         if(desiredUnit == 'em') {
449             if(IS_COMPUTED_STYLE_SUPPORTED)
450                 return UnitFontAdapter.em || (UnitFontAdapter.em = new UnitFontAdapter( "em", 1 ));

463         }

483     /**
484      * @requires nonPixelExp
485      * @param {HTMLElement} el
486      * @param {Array} String[] of matches from nonPixelExp.exec( val ).
487      */
488     function convertNonPixelToPixel(el, matches) {
489 
490         if(el.runtimeStyle) {
491 
492             // http://erik.eae.net/archives/2007/07/27/18.54.15/#comment-102291
493             // If we're not dealing with a regular pixel number
494             // but a number that has a weird ending, we need to convert it to pixels.
495 
496             var val = matches[0]; // grab the -1.2em or whatever.
497             if(parseFloat(val) == 0) {
498                 return "0px";
499             }
500 
501             var s = el[style],
502                 sLeft = s.left,
503                 rs = el.runtimeStyle,
504                 rsLeft = rs.left;
505 
506             rs.left = el[currentStyle].left;
507             s.left = (val || 0);
508 
509             // The element does not need to have position: to get values.
510             // IE's math is a little off with converting em to px; IE rounds to 
511             // the nearest pixel.
512             val = s.pixelLeft + px;
513             // put it back.
514             s.left = sLeft;
515             rs.left = rsLeft;
516             return val;
517         }
518     }

577     /** 
578      * Used for converting getComputedStyle(el,"") 
579      * value to an em value. If the property is fontSize, 
580      * it is necessary to get the parent's computed fontSize.
581      * @memberOf APE.dom
582      * @param {HTMLElement} parentNode
583      * @param {String} prop style property to work with.
584      * @param {String} val the value, such as "12.3px".
585      * @param {ComputedCSSStyleDeclaration} computedStyle object to work with.
586      */
587     function convertPixelToEmEx(parentNode, prop, val, computedStyle) {
588         val = this.val || val;
589         var match = pxExp.exec(val),
590             defaultView = parentNode.ownerDocument.defaultView;
591         if(match) {
592             if(match[0]) {
593                 var fontSize,
594                     d = parseFloat(val);
595 
596                 if(!this.fontSize) {
597                     // If we're trying to get fontSize
598                     if(prop == "fontSize") {
599                         fontSize = defaultView[getCS](parentNode,'').fontSize;
600                     }
601                     else
602                         fontSize = computedStyle.fontSize;
603                 }
604                 fontSize = parseFloat(fontSize||this.fontSize);
605                 if(isFinite(d)) {
606                     return d/fontSize * this.fontSizeMultiplier + this.unit;
607                 }
608             }
609         }
610         if(!val) return "";
611         else if(isFinite(val)) return val + this.unit;
612         return val; // return input.
613     }

627     /** 
628      * Finds the containing block of el, as per CSS 2.1 sec 10.1
629      * @memberOf APE.dom 
630      * @param {HTMLElement} el
631      * @return {HTMLElement} el's containing block.
632      */
633     function getContainingBlock(el) {
634         var elPosition = _getComputedStyle(el, "position"), 
635             docEl = el.ownerDocument.documentElement,
636             parent = el.parentNode;
637         if(/^(?:r|s)/.test(elPosition) || !elPosition) return parent;
638         if(elPosition == "fixed") return null;
639         while(parent && parent != docEl) {
640             if(_getComputedStyle(parent, "position") != "static")
641                 return parent;
642             parent = parent.parentNode;
643         }
644         return docEl;
645     }
