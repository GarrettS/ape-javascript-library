 10 APE.namespace("APE.dom");
 11 (function() {
 12     APE.mixin(APE.dom, 
 13         /** @scope APE.dom */{
 14         hasToken : hasToken,
 15         removeClass : removeClass,
 16         addClass : addClass,
 17         getElementsByClassName : getElementsByClassName,
 18         findAncestorWithClass : findAncestorWithClass
 19     });
 20 
 21     /** @param {String} s string to search
 22      * @param {String} token white-space delimited token the delimiter of the token.
 23      * This is generally used with element className: 
 24      * @example if(dom.hasToken(el.className, "menu")) // element has class "menu".
 25      */
 26     function hasToken (s, token) {
 27         return getTokenizedExp(token,"").test(s);
 28     }
 29 
 30     /** @param {HTMLElement} el  
 31      * @param {String} klass className token(s) to be removed. 
 32      * @description removes all occurances of <code>klass</code> from element's className.
 33      */
 34     function removeClass(el, klass) {
 35         el.className = normalizeString(el.className.replace(getTokenizedExp(klass, "g")," "));
 36     }
 37     /** @param {HTMLElement} el  
 38      * @param {String} klass className token(s) to be added. 
 39      * @description adds <code>klass</code> to the element's class attribute, if it does not 
 40      * exist.
 41      */
 42     function addClass(el, klass) {
 43         if(!el.className) el.className = klass;
 44         if(!getTokenizedExp(klass).test(el.className)) el.className += " " + klass;
 45     }
 46     
 47     var Exps = { };
 48     function getTokenizedExp(token, flag){
 49         var p = token + "$" + flag; 
 50         return (Exps[p] || (Exps[p] = RegExp("(?:^|\\s)"+token+"(?:$|\\s)", flag)));
 51     }
 52 
 53     /** @param {HTMLElement} el  
 54      * @param {String} tagName tagName to be searched. Use "*" for any tag.
 55      * @param {String} klass className token(s) to be added. 
 56      * @return {Array} Elements with the specified tagName and className.
 57      * Searches will generally be faster with a smaller HTMLCollection
 58      * and shorter tree.
 59      */
 60     function getElementsByClassName(el, tagName, klass){
 61         var ret = Array(len);
 62 
 63         if(el.getElementsByClassName && (tagName === "*" || !tagName)) {
 64             // Native performance boost.
 65             return ret.slice.call(el.getElementsByClassName(klass));
 66         }
 67         var exp = getTokenizedExp(klass,""),
 68             collection = el.getElementsByTagName(tagName),
 69             len = collection.length,
 70             counter = 0;
 71         for(i = 0; i < len; i++){
 72             if(exp.test(collection[i].className))
 73                 ret[counter++] = collection[i];
 74         }
 75         ret.length = counter; // trim array.
 76         return ret;
 77     }
 78 
 79     /** Finds an ancestor with specified className
 80      */
 81     function findAncestorWithClass(el, klass) {
 82         if(el == null)
 83             return null;
 84         var exp = getTokenizedExp(klass,"");
 85         for(var parent = el.parentNode;parent != null;){
 86             if( exp.test(parent.className) )
 87                 return parent;
 88             parent = parent.parentNode;
 89         }
 90         return null;
 91     }
 92 
 93 var STRING_TRIM_EXP = /^\s+|\s+$/g,
 94     WS_MULT_EXP = /\s\s+/g;
 95 function normalizeString(s) { return s.replace(STRING_TRIM_EXP,'').replace(WS_MULT_EXP, " "); }
 96 
 97 if(!document.getElementsByClassName) nativeGEBCN = null;
 98 })();
