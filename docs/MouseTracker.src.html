1 APE.MouseTracker = {
  2 
  3     subscribers : { },
  4 
  5     activeObject : null,

  7     register : function(subscriber, id) {
  8         id = id || subscriber.id;
  9 
 10         if(!this.subscribers.hasOwnProperty(id)) 
 11             this.subscribers[id] = [ ];
 12         if(!this.inited)
 13             this.init();
 14         var stack = this.subscribers[id],
 15             i = 0, len = stack.length;
 16         while(i < len) {
 17             if(subscriber = stack[i++]) return false;
 18         }
 19         stack[i] = subscriber;
 20         return true;
 21     },
 22 
 23     unregister : function(subscriber, id) {
 24         id = id || subscriber.id;
 25         var x = this.subscribers[id],
 26             i = 0,
 27             len;
 28         if(x) {
 29             len = x.length;
 30             while(i++ < len) {
 31                 if(x[i] === subscriber)
 32                     x.splice(i, 1);
 33             }
 34         }
 35         return x;
 36     },
 37 
 38     init : function() {
 39         
 40         var d = document,
 41             docEl = d.documentElement,
 42             ds = docEl.style;
 43         APE.EventPublisher.add(d, "onmousedown", this.mouseDown, this);
 44         APE.EventPublisher.add(d, "onkeypress", this.keyPressed, this);
 45         APE.EventPublisher.add(d, "onmousemove", this.mouseMove, this);
 46         APE.EventPublisher.add(d, "onmouseup", this.mouseUp, this);
 47 
 48         // prevent text selection while dragging.
 49         if('onselectstart' in d) {
 50             APE.EventPublisher.get(d, "onselectstart").addBefore(this.hasActiveObject, this);
 51         }
 52         else {
 53             APE.EventPublisher.get(d, "onmousedown").addAfter(this.preventUserSelection, this);
 54             APE.EventPublisher.get(d, "onmouseup").addAfter(this.preventUserSelection, this);
 55         }
 56         this.inited = true;
 57 
 58         this.userSelectType = "MozUserSelect"in ds ? "MozUserSelect" : 
 59             "KhtmlUserSelect"in ds ? "KhtmlUserSelect" : "userSelect"in ds ? "userSelect" : "";
 60     },
 61 
 62     hasActiveObject : function() {
 63         return this.activeObject != null;
 64     },
 65 
 66     preventUserSelection : function() {
 67         document.documentElement.style[this.userSelectType] = (this.activeObject ? "none" : "");
 68     },
 69 
 70     mouseDown : function(e) { 
 71     // TODO: This is too complicated. Focus/selection. DragMultiple. draggableList. 
 72     // Need a way to encapsulate those complexities.
 73         if(!e) 
 74             e = event;
 75             
 76         var APE = window.APE,
 77             dom = APE.dom,
 78             target = dom.Event.getTarget(e),
 79             activeObject = null,
 80             instances = APE.MouseTracker.subscribers,
 81             i = 0, len;
 82 
 83         for(var testNode = target;activeObject == null && testNode != null;
 84                             testNode = dom.findAncestorWithAttribute(testNode, "id")) {
 85             if(testNode !== null) {
 86                 activeObject = instances[testNode.id];
 87             }
 88         }
 89         
 90         if(activeObject) { // found. 
 91 
 92             this.mousedownX = ('pageX'in e ? e.pageX : e.clientX + scrollOffsets.left);
 93             this.mousedownY = ('pageY'in e ? e.pageY : e.clientY + scrollOffsets.top);
 94             for(var len = activeObject.length; i < len; i++) {
 95                 if(activeObject[i].mouseDown(e))
 96                     this.activeObject = activeObject[i];
 97             }
 98         }
 99     },
100 
101     getEventCoords : APE.dom.Event.getCoords,
102 
103     mouseMove : function(e) {
104 
105         var activeObject = this.activeObject;
106         
107         if(!e)
108             e = event;
109         
110         if(activeObject == null) return;
111  
112         var eventCoords = this.getEventCoords(e),
113             ePageX = eventCoords.x, ePageY = eventCoords.y,
114             distX = ePageX - this.mousedownX,
115             distY = ePageY - this.mousedownY;
116 
117         activeObject.newX = activeObject.origX + distX;
118         activeObject.newY = activeObject.origY + distY;
119         activeObject.mouseMove(e);
120     },
121 
122     mouseUp : function(e) {
123 
124         if(this.activeObject == null) {
125             return;
126         }
127     // IE will fire this event twice when mouse was held.
128 
129         if(!e)
130             e = event;
131         
132         var activeObject = this.activeObject;
133         if(activeObject.mouseUp(e))
134             this.activeObject = null;
135     },

137     keyPressed : function(){}
