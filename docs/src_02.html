 10 /**
 11  * @class <p>
 12  * steals the event handler and creates a callStack. 
 13  * that fires in its place.
 14  * </p><p>
 15  * an EventPublisher itself publishes beforeFire and afterFire.
 16  * This makes it possible to add AOP before advice to the callStack.
 17  * </p><p>
 18  * adding before-before advice is possible, but will impair performance.
 19  * Instead, add multiple beforeAdvise with: publisher.addBefore(fp, thisArg).add(fp2, thisArg);
 20  * </p><p>
 21  * Adding beforeEach and afterEach, to create advice for each callback would require modification 
 22  * to the registry (see comments below). I have not yet found a real need for this.
 23  * </p>
 24  */
 25 /**
 26  * @constructor
 27  */
 28 APE.EventPublisher = function(src, type) {
 29 	this.src = src;
 30 	// Really could use a List of bound methods here. 
 31 	this._callStack = [];
 32 	this.type = type;
 33 };
 34 
 35 APE.EventPublisher.prototype = {
 36 
 37 /**  
 38  *  @param {Function} fp the callback function that gets called when src[sEvent] is called.
 39  *  @param {Object} thisArg the context that the function executes in.
 40  */
 41 	add : function(fp, thisArg) {
 42 		this._callStack.push([fp, thisArg||this.src]);
 43 		return this;
 44 	}
 45 /**  Adds beforeAdvice to the callStack. This fires before the callstack. 
 46  *  @param {Function} fp the callback function that gets called when src[sEvent] is called.
 47  *  @param {Object} thisArg the context that the function executes in.
 48  *  @return {boolean} proceed false stops the callstack and returns false to the original call.
 49  */
 50 	,addBefore : function(f, thisArg) {
 51 		return APE.EventPublisher.add(this, "beforeFire", f, thisArg); 
 52 	}
 53 	
 54 /**  Adds afterAdvice to the callStack. This fires after the callstack. 
 55  *  @param {Function} fp the callback function that gets called when src[sEvent] is called.
 56  *  @param {Object} thisArg the context that the function executes in.
 57  *  @return {boolean} return false tor return false to the original call.
 58  */
 59 	,addAfter : function(f, thisArg) {
 60 		return APE.EventPublisher.add(this, "afterFire", f, thisArg); 
 61 	}
 62 
 63 	/** 
 64 	 * @param {String} "beforeFire", "afterFire" conveneince.
 65 	 */
 66 	,getEvent : function(type) {
 67 		return APE.EventPublisher.get(this, type);
 68 	}
 69 
 70 	,remove : function(fp, thisArg) {
 71 		var cs = this._callStack;
 72 		if(!thisArg) thisArg = this.src;
 73 		for(var i = 0, len = cs.length; i < len; i++) {
 74 			var call = cs[i];
 75 			if(call[0] === fp && call[1] === thisArg) {
 76 				return cs.splice(i, 1);
 77 			}
 78 		}
 79 		return null;
 80 	}
 81 
 82 	,removeBefore : function(fp, thisArg) {
 83 		return this.getEvent("beforeFire").remove(fp, thisArg);
 84 	}
 85 
 86 	,removeAfter : function(fp, thisArg) {
 87 		return this.getEvent("afterFire").remove(fp, thisArg);
 88 	}
 89 
 90 	,toString : function() {
 91 		return  "APE.EventPublisher: {src=" + this.src + ", type=" + this.type +
 92 			 ", length="+this._callStack.length+"}";
 93 	}
 94 };
 95 
 96 /** 
 97  *  @static
 98  *  @param {Object} src the object which calls the function
 99  *  @param {String} sEvent the function that gets called.
100  *  @param {Function} fp the callback function that gets called when src[sEvent] is called.
101  *  @param {Object} thisArg the context that the function executes in.
102  */
103 APE.EventPublisher.add = function(src, sEvent, fp, thisArg) {
104 	return APE.EventPublisher.get(src, sEvent).add(fp, thisArg);
105 };
106 
107 /** 
108  * @static
109  * @private.
110  * @return {boolean} false if any one of callStack's methods return false.
111  */
112 APE.EventPublisher.fire = function(publisher) {
113 	// This closure sucks. We should have partial/bind in ES.
114 	// If we did, this could more reasonably be a prototype method.
115 	
116 	// return function w/identifier doesn't work in Safari 2.
117 	return fireEvent; 
118 	function fireEvent(e) {
119 		var preventDefault = false;
120 
121 		// beforeFire can affect return value.
122 		if(typeof publisher.beforeFire == "function") {
123 			if(publisher.beforeFire(e) == false)
124 				preventDefault = true;
125 		}
126 
127 		var cs = publisher._callStack;
128 		for(var i = 0, len = cs.length; i < len; i++) {
129 			var csi = cs[i]; 
130 			// If an error occurs, continue the event fire,
131 			// but still throw the error.
132 			try {
133 				// TODO: beforeEach to prevent or advise each call.
134 				if(csi[0].call(csi[1], e) == false)
135 					preventDefault = true; // continue main callstack and return false afterwards.
136 				// TODO: afterEach
137 			}
138 			catch(ex) {
139 				APE.deferError(ex);
140 			}
141 		}
142 		// afterFire can prevent default.
143 		if(typeof publisher.afterFire == "function") {
144 			if(publisher.afterFire(e) == false)
145 				preventDefault = true;
146 		}
147 		return !preventDefault;
148 	}
149 };

151 /** 
152  * @static
153  * @param {Object} src the object which calls the function
154  * @param {String} sEvent the function that gets called.
155  *
156  * Looks for an APE.EventPublisher in the Registry.
157  * If none found, creates and adds one to the Registry.
158  */
159 APE.EventPublisher.get = function(src, sEvent) {
160 
161 	var publisherList = this.Registry.hasOwnProperty(sEvent) && this.Registry[sEvent] || (this.Registry[sEvent] = []);
162 	
163 	for(var i = 0, len = publisherList.length; i < len; i++)
164 		if(publisherList[i].src === src)
165 			return publisherList[i];
166 	
167 	// not found.
168 	var publisher = new APE.EventPublisher(src, sEvent);
169 	// Steal. 
170 	if(src[sEvent])
171 		publisher.add(src[sEvent], src);
172 	src[sEvent] = this.fire(publisher);
173 	publisherList.push(publisher);
174 	return publisher;
175 };
176 
177 /** 
178  * Map of [APE.EventPublisher], keyed by type.
179  * @private
180  * @static
181  */
182 APE.EventPublisher.Registry = {};
183 
184 /**
185  * @static
186  * called onunload, automatically. Cleans up all 
187  */
188 APE.EventPublisher.cleanUp = function() {
189 	for(var type in this.Registry) {
190 		var publisherList = this.Registry[type];
191 		for(var i = 0, len = publisherList.length; i < len; i++) {
192 			var publisher = publisherList[i];
193 			publisher.src[publisher.type] = null;
194 		}
195 	}
196 };
197 
198 APE.EventPublisher.get( window, "onunload" ).addAfter( APE.EventPublisher.cleanUp, APE.EventPublisher );
