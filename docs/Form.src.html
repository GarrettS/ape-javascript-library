1 /**
  2  * @fileoverview
  3  * Form
  4  *
  5  * For serializing a Form for use with Ajax.
  6  *
  7  * Disabled elements and elements with no name are not serialized.
  8  * Select multiple and checkboxes or any form elements with the same name are supported.
  9  * 
 10  * @author Garrett Smith
 11  */
 12 APE.namespace("APE.form");
 13 
 14 /**
 15  * @constructor
 16  * Creates a Form wrapper that can be used to serialize the data 
 17  * on submit.
 18  * @param {HTMLFormElement} form the form to be used. A fieldset or other element that contains 
 19  * form elements may also be passed in.
 20  * 
 21  */
 22 APE.form.Form = function(form) {
 23     this.form = form;
 24 };
 25 
 26 APE.form.Form.prototype = {
 27 
 28     /**
 29      * Serializes the form according to HTML 4.01
 30      * http://www.w3.org/TR/html401/interact/forms.html#successful-controls
 31      *
 32      * @param {HTMLInputElement} [submit] input that needs to be included in 
 33      * the successful controls.
 34      * @return {Object} An object whose property names are element names and 
 35      * values are arrays.
 36      */
 37     toObject : function(submit) {
 38         var form = this.form,
 39            elements = form.elements, i, len, 
 40            target,
 41            element, type, name, ontype = /^(?:rad|che)/,
 42            json = {},
 43            options,
 44            option,
 45            j, jlen,
 46            p, plen,
 47            par, sib, doc, node,
 48             // Although no browsers actually include "image" 
 49             // in elements collection.
 50            submitTypeExp = /^(?:submit|image)$/;
 51 
 52         // If no elements property, it is not a form.
 53         // Create a form and grab the elements off it.
 54         if(!elements) {
 55             node = form;
 56             par = node.parentNode;
 57             sib = node.nextSibling;
 58             doc = node.ownerDocument||document;
 59             form = doc.createElement('form');
 60             form.appendChild(node);
 61             elements = []; // IE 8 can't slice(elements).
 62             len = form.elements.length;
 63             for(i = 0; i < len; i++) 
 64                 elements[i] = form.elements[i];
 65             par.insertBefore(node, sib);
 66         }
 67 
 68         len = elements.length;
 69         for(i = 0; i < len; i++) {
 70             element = elements[i];
 71             type = element.type;
 72             name = element.name;
 73             
 74             // Skip unsuccessful controls.
 75             if(!name || element.disabled || type == "reset" 
 76               || type == "button" || submitTypeExp.test(type) // only on event.
 77               || (ontype.test(type) && !element.checked)
 78               || (type == "object" && type.declared) 
 79                   // IE can't get value from select-multiple when option value attr is empty.
 80               || type == "select-multiple" && element.selectedIndex == -1) continue;
 81           
 82             p = json[name];
 83 
 84             if(!p) json[name] = p = [];
 85             plen = p.length;
 86 
 87             options = element.options;
 88             if(options) {
 89                 if(type == "select-multiple") {
 90                     for(j = 0, jlen = options.length; j < jlen; j++) {
 91                         option = options[j];
 92                         if(option.selected && !option.disabled) {
 93                             p[plen++] = option.value || option.text;
 94                         }
 95                     }
 96                 }
 97                 else {
 98                     option = options[element.selectedIndex];
 99                     if(option && !option.disabled)
100                         p[plen] = option.value || option.text;
101                 }
102             }
103         
104           // https://bugzilla.mozilla.org/show_bug.cgi?id=371432
105           // http://www.w3.org/TR/file-upload/
106           //
107           // Just take the first file.
108             else if(type == "file") { 
109                 var file = element.files;
110                 file = file && file[0];
111                 if(file) {
112                     p[plen] = file;
113                 }
114             }
115             else {
116                 p[plen] = element.value;
117             }
118         }
119 
120         name = !!submit && submit.name;
121         if(name) {
122             if(!json[name]) p = json[name] = [];
123             p[plen++] = submit.value;
124         }
125         return json;
126     },
127 
128     /**
129      * @return {Array} array of strings for successful data.
130      * @param {HTMLInputElement} [submit] input that needs to be included.
131      */
132     getMultipartFormData : function(submit) {
133     // Doesn't encode data.
134     // http://groups.google.com/group/comp.lang.javascript/browse_thread/thread/eada69993b5ae08a/5943a32b5ecca6e6?lnk=gst&q=encodeURIComponent+post+xhr#5943a32b5ecca6e6
135     // http://www.devx.com/Java/Article/17679/1954
136     var json = this.toObject(submit), prop, value, file, i, len, result = [],
137         nn = '\r\n\r\n';
138     for(prop in json) {
139         value = json[prop];
140         for(i = 0, len = value.length; i < len; i++) {
141             file = value[i];
142             if(!file) continue;
143             result[result.length] = "Content-Disposition: form-data; " +
144               'name="'+prop +'";'
145               + (file.fileName && file.getAsBinary ? ' filename="'+file.fileName +'"'
146                 + nn + file.getAsBinary() : nn + file);
147             }
148         }
149         return result;
150     },
151 
152     /** 
153      * @return {String} the query string, including the "?".
154      * @param {HTMLInputElement} [submit] button that was clicked.
155      */
156     getQueryString : function(submit) {
157         var json = this.toObject(submit), prop, i, value, encodedProp, result = [], ws = /%20/g;
158         for(prop in json) {
159             value = json[prop];
160             encodedProp = encodeURIComponent(prop);
161             for(i = 0, len = value.length; i < len; i++) {
162                 result[result.length] = encodedProp +
163                   "=" + encodeURIComponent(value[i]).replace(ws,'+');
164             }
165         }
166         return result.join("&");
167     },
168 
169     /** 
170      * Builds a query string for GET or POST.
171      * @return {String} GET or POST data.
172      * @throws {Error} if the enctype is "multipart/form-data"
173      * @param {HTMLInputElement} [submit] input that needs to be included.
174      */
175     serialize : function(submit) {
176         var method = this.form.method.toLowerCase();
177         if(method == "get") {
178             return this.form.action + "?" + this.getQueryString(submit);  
179         }
180         if(method == "post") {
181             if(this.form.enctype == "multipart/form-data")
182                 throw Error("multipart/form-data: Use getMultipartFormData()");
183             return this.getQueryString(submit);
184         }
185     }
186 };
