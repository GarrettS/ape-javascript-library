 15 /** 
 16  * @class 
 17  * @extends APE.anim.Animation 
 18  * @base APE.anim.Animation
 19  * @see APE.anim.Transitions
 20  * @constructor
 21  * @param {String} id of the element
 22  * @param {Object} styleObject in the form of {color: "#030", fontSize : "12px"}
 23  * @param {ufloat} [duration] optional number of seconds.
 24  * @param {Function} [transition] optional funtion that takes a float [0-1] and returns a float [0-1]
 25  */
 26 APE.anim.StyleTransition = function(id, styleObject, duration, transition) {
 27 	this.id = id;
 28 	this.adapters = [];
 29 	this.el = document.getElementById(this.id);
 30 	this.transition = transition;
 31 	this.init(styleObject);
 32 };
 33 
 34 APE.extend(APE.anim.StyleTransition, APE.anim.Animation, 
 35 	new function() {
 36 	/**@ignore*/
 37 	var dom = APE.dom;
 38 
 39 	this.inited = false;
 40 
 41 	/**
 42 	 * @ignore
 43 	 * see TransitionAdapter 
 44 	 */
 45 	this.init = function(styleObject) {
 46 		if(this.inited) return;
 47 
 48 		var adapters = [];
 49 
 50 		// Loop through style object to find values.
 51 		for(var prop in styleObject) {
 52 			var toValue = styleObject[prop];
 53 			if(!toValue) continue; // CSSStyleRule.
 54 
 55 			var fromValue = dom.findInheritedStyle(this.el, prop);
 56 
 57 			if(prop == "opacity" && !("opacity"in this.el.style)
 58 								&& ("filter"in this.el.style)) {
 59 				prop = "alpha";
 60 				fromValue = dom.getFilterOpacity(this.el);
 61 			}
 62 	
 63 			var adapter = TransitionAdapter.fromValues(prop, fromValue, toValue);
 64 			adapters.push( adapter );
 65 		}
 66 		this.adapters = adapters;
 67 		var el = document.getElementById(this.id);
 68 		var transition = this.transition;
 69 		var self = this;
 70 
 71 		/** Runs the animation, getting the correct value  
 72 		 * from each ITransitionAdapter.
 73 		 * This run() method gets called by Animation.
 74 		 * @memberOf APE.anim.StyleTransition
 75 		 */
 76 		this.run = function run(rationalValue) {
 77 			var newValue = transition ? transition(rationalValue) : rationalValue;
 78 			for(var i = 0, len = adapters.length; i < len; i++) {
 79 				var adapter = adapters[i];
 80 				el.style[adapter.prop] = adapter.blendTo(newValue);
 81 			}
 82 		};
 83 	};
 84 
 85 	this.toggleDirection = function() {
 86 		if(!this.started) {
 87 			this.start();
 88 			return;
 89 		}
 90 		if(this.isReversed)
 91 			this.seekTo(1);
 92 		else {
 93 			this.seekTo(0);
 94 		}
 95 	};
 96 
 97 	var ColorRGB = APE.color.ColorRGB;
 98 
 99 	/*
100 	 * Factory for TransitionAdapter Interface.
101 	 * @return ITransitionAdapter for a specific type of style setting
102 	 * during run.
103 	 * The ITransitionAdapter implements blendTo(rationaValue);.
104 	 */
105 	var TransitionAdapter = {
106 		lengthExp : /([\d]+)(%|px|pt|em|pc|ex)$/i,
107 		colorExp : /color/i, 
108 		positiveLengthExp : /width|height|padding/i, 
109 		filterExp : /alpha/,
110 		opacityExp : /^opacity/,
111 
112 		fromValues : function(prop, fromValue, toValue) {
113 
114 			if(this.positiveLengthExp.test( prop )) 
115 				return new PositiveLengthTransitionAdapter(prop, fromValue, toValue);
116 
117 			if(this.colorExp.test( prop )) 
118 				return new ColorTransitionAdapter(prop, fromValue, toValue);
119 
120 			if(this.lengthExp.test( prop )) 
121 				return new LengthTransitionAdapter(prop, fromValue, toValue);
122 
123 			if(this.filterExp.test( prop )) 
124 				return new FilterTransitionAdapter(prop, fromValue, toValue);
125 
126 			if(this.opacityExp.test( prop )) 
127 				return new OpacityTransitionAdapter(prop, fromValue, toValue);
128 
129 			// Return an object that sets toValue on completion.
130 			return new NullTransitionAdapter(prop, fromValue, toValue);
131 		}
132 	};
133 
134 	/* Adapter/Strategy interface.
135 	 * blendTo - returns a string containing the blended values
136 	 */
137 	function ColorTransitionAdapter(prop, fromValue, toValue) {
138 		this.prop = prop;
139 		this.fromValue = ColorRGB.fromString(fromValue);
140 		this.toValue = ColorRGB.fromString(toValue);
141 
142 		// This is where we mix fromValue and toValue,
143 		// to avoid the creation of new ColorRGB for each frame.
144 		this.currentValue = new ColorRGB();
145 	}
146 
147 	ColorTransitionAdapter.prototype = {
148 		blendTo : function(rationalValue) {
149 			var c = ColorRGB.blend(this.fromValue, this.toValue, rationalValue, this.currentValue).toString();
150 			return c;
151 		},
152 		toString : function() {
153 			return"ColorTransitionAdapter: " + this.prop + ", " + this.fromValue + ", " + this.toValue; 
154 		}
155 	};
156 
157 	function LengthTransitionAdapter(prop, fromValue, toValue) {
158 		this.prop = prop;
159 		var lengthExp = TransitionAdapter.lengthExp;
160 		var fromValues = lengthExp.exec(fromValue);
161 		var toValues = lengthExp.exec(toValue);
162 
163 		this.fromValue = parseInt(fromValues[0]);
164 		this.units = fromValues[2];
165 		var toValues = lengthExp.exec(toValue);
166 		this.toValue = parseInt(toValues[0]);
167 		if(this.units != toValues[2]) {
168 			throw Error("Can't convert " + this.units + " to " + toValues[2]);
169 		}
170 	}
171 
172 	LengthTransitionAdapter.prototype = {
173 		blendTo : function(rationalValue) {
174 			return ((this.fromValue * inverse) + (this.toValue * rationalValue)) + this.units;
175 		},
176 		toString : function() {
177 			return"LengthTransitionAdapter: " +this.prop + ", " + this.fromValue + ", " + 
178 				this.toValue  + ", " + this.units;
179 		}
180 	};
181 
182 	function PositiveLengthTransitionAdapter() {
183 		LengthTransitionAdapter.apply(this, arguments);
184 	}
185 
186 	/** 
187 	 * @extends LengthTransitionAdapter
188 	 */
189 	APE.extend(PositiveLengthTransitionAdapter, LengthTransitionAdapter);
190 
191 	PositiveLengthTransitionAdapter.prototype.blendTo = function(rationalValue) {
192 		var inverse = 1-rationalValue;
193 		var v = Math.max((this.fromValue * inverse) + (this.toValue * rationalValue), 0) + this.units;
194 		return v;
195 	};
196 
197 	function OpacityTransitionAdapter(prop, fromValue, toValue) {
198 		this.prop = prop;
199 		this.fromValue = parseFloat(fromValue);
200 		this.toValue = parseFloat(toValue);
201 	}
202 
203 	/*
204 	 * constructor OpacityTransitionAdapter
205 	 * For browsers that don't have filters.
206 	 */
207 	OpacityTransitionAdapter.prototype = {
208 		blendTo : function(rationalValue) {
209 			var inverse = 1-rationalValue;
210 			// If opacity > 1, the browser treats it the same as 1. Let native code handle max(val, 1);
211 			var v = Math.max((this.fromValue * inverse) + (this.toValue * rationalValue), 0);
212 			return v;
213 		},
214 
215 		toString : function() {
216 			return"OpacityTransitionAdapter: " + this.fromValue + ", " + this.toValue;
217 		}
218 	};
219 
220 	function FilterTransitionAdapter(prop, fromValue, toValue) {
221 		this.prop = "filter";
222 		this.fromValue = fromValue;
223 		this.toValue = toValue;
224 	}
225 	/* 
226 	 * constructor FilterTransitionAdapter
227 	 * performs adapter service using IE filters crap. 
228 	 */
229 	FilterTransitionAdapter.prototype = {
230 		blendTo : function(rationalValue) {
231 			var inverse = 1-rationalValue;
232 			var v = Math.abs((this.fromValue * inverse) + (this.toValue * rationalValue), 0);
233 			return"alpha(opacity=" + Math.abs(v * 100) + ")";
234 		},
235 
236 		toString : function() {
237 			return"FilterTransitionAdapter: " + this.fromValue + ", " + this.toValue;
238 		}
239 	};
240 
241 	/* 
242 	 * constructor NullTransitionAdapter
243 	 * This is sort of a Null Object. It does implement blendTo,
244 	 * setting the toValue at the very end.
245 	 */
246 	function NullTransitionAdapter(prop, fromValue, toValue) {
247 		this.prop = prop;
248 		this.toValue = toValue;
249 		this.fromValue = fromValue;
250 	}
251 	
252 	NullTransitionAdapter.prototype = {
253 		blendTo : function(rationalValue){
254 			if(rationalValue == 1) 
255 				return this.toValue;
256 			return this.fromValue;
257 		},
258 		toString : function(){ 
259 			return"NullTransitionAdapter: can't perform a smooth transition with: " +this.prop;
260 		}
261 	};
262 });
