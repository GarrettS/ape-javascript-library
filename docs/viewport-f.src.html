  4 APE.namespace("APE.dom");
  5 
  6 (function() {
  7 
  8     // Public exports.
  9     APE.mixin(APE.dom, {
 10         getScrollOffsets : getScrollOffsets,
 11         getViewportDimensions : getViewportDimensions
 12     });
 13 
 14     var docEl = document.documentElement,
 15         IS_BODY_ACTING_ROOT = docEl && docEl.clientWidth === 0;
 16     docEl = null;
 17 
 18     /** @memberOf APE.dom 
 19      * @name getScrollOffsets
 20      * @function
 21      * @return an object with <code>width</code> and <code>height</code>.
 22      * This will exhibit a bug in Mozilla, which is often 5-7 pixels off.
 23      */
 24     function getScrollOffsets() {
 25         var f;
 26         if("pageXOffset"in window)
 27             f = function() {
 28                 return{ left:window.pageXOffset, top: window.pageYOffset};
 29             };
 30         else if(IS_BODY_ACTING_ROOT)
 31             f = function() {
 32                 var body = document.body;
 33                 return{ left: body.scrollLeft, top : body.scrollTop};
 34             };
 35         else f = function() {
 36             var docEl = document.documentElement;
 37             return{ left : docEl.scrollLeft, top : docEl.scrollTop };
 38         };
 39         this.getScrollOffsets = f;
 40         return f();   
 41     }
 42 
 43     /** @memberOf APE.dom 
 44      * @name getViewportDimensions
 45      * @function
 46      * @return an object with <code>width</code> and <code>height</code>.
 47      * This will exhibit a bug in Mozilla, which is often 5-7 pixels off.
 48      */
 49     function getViewportDimensions() {
 50         var f;
 51 
 52         if(typeof document.width == "number") {
 53             f = function(){
 54                 var d = document, docEl = d.documentElement;
 55                 return{ width : Math.min(d.width, docEl.clientWidth), 
 56                     height : Math.min(d.height, docEl.clientHeight)};
 57             };
 58         } else if("innerWidth"in window) {
 59             f = function() {
 60                 return { width: window.innerWidth, height: window.innerHeight };
 61             };
 62         }
 63         else if(IS_BODY_ACTING_ROOT) {
 64             f = function() {
 65                 var body = document.body;
 66                 return { width: body.clientWidth, height: body.clientHeight};
 67             };
 68         }
 69         else f = function(){
 70             return{width:docEl.clientWidth, height:docEl.clientHeight};
 71         };
 72         this.getViewportDimensions = f;
 73 
 74         // Call in correct global context (for iframe).
 75         return f(); 
 76     }
 77 })();
