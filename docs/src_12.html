1 //------------------------ dom Location Functions -----------------------------------
  2 APE.namespace("APE.dom");
  3 
  4 (function(){
  5 	
  6 	var dom = APE.dom;
  7 
  8 	dom.findAncestorWithAttribute = function(el, attrName, value) {
  9 		for(var parent = el.parentNode;parent != null;){
 10 			if(parent[attrName])
 11 				if(parent[attrName] == value || value == "*")
 12 					return parent;			
 13 			parent = parent.parentNode;
 14 		}
 15 		return null;
 16 	};
 17 
 18 	dom.findAncestorWithTagName = function(el, tagName) {
 19 		if(!el.parentNode) return null;
 20 		el = el.parentNode;
 21 		return el.tagName == tagName ? el : findAncestorWithTagName(el, tagName);
 22 	};
 23 
 24 	dom.findNextSiblingElement = function(el) {
 25 		for(var ns = el.nextSibling; ns != null; ns = ns.nextSibling)
 26 			if(ns.tagName) 
 27 				return ns;
 28 			return null;
 29 	};
 30 
 31 	dom.findPreviousSiblingElement = function(el) {
 32 		for(var ps = el.previousSibling; ps != null; ps = ps.previousSibling)
 33 			if(ps.tagName) 
 34 				return ps;
 35 			return null;
 36 	};
 37 
 38 	dom.getChildElements = function(el) {	
 39 		var ret = [];
 40 		var cn = el.childNodes;
 41 		for(var i = 0, len = cn.length; i < len; i++) {
 42 			if(cn[i].tagName)
 43 				ret[ret.length] = cn[i];
 44 		}
 45 		return ret;
 46 	};
 47 
 48 	dom.getContainingBlock = function(el) {
 49 
 50 		var root = document.documentElement;
 51 		var elPosition = getStyle(el, "position");
 52 		if(elPosition == "relative" || elPosition == "static" || !elPosition) return el.parentNode;
 53 		for(var parent = el.parentNode; parent != null && parent != root; ) {
 54 			if(getStyle(parent, "position") != "static")
 55 				return parent;
 56 			parent = parent.parentNode;
 57 		}
 58 		return root;
 59 	};
 60 
 61 	/** Returns true if a contains b.
 62 	 */
 63 	dom.contains = function(el, b) {
 64 		while(el != b && (b = b.parentNode) != null);
 65 		return el === b;
 66 	};
