  9 var dom = APE.dom;

 11     dom.Event = {},{
 12         getTarget : getTarget,

 15 
 16 function getTarget(e) { return e && e.target || event.srcElement; }
 17 function getCoords(e) {
 18     var f;
 19     if("pageX"in e) {
 20         f = function(e) {
 21             return {
 22                 x : e.pageX,
 23                 y : e.pageY
 24             };
 25         };
 26     }
 27     else {
 28         f = function(e){
 29             var scrollOffsets = dom.getScrollOffsets(), e = e||event;
 30             return {
 31                 x : e.clientX + scrollOffsets.left,
 32                 y : e.clientY + scrollOffsets.top
 33             }
 34         };
 35     }
 36     return(dom.Event.getCoords = f)(e);
 37 }
 38 })();
