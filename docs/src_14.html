1 /** Locking aspect. 
  2  * @author Garrett Smith
  3  *
  4  * Can be used for AsyncRequest, Animation, or anything that 
  5  * involves setTimeout or setInterval.
  6  * @aspect
  7  * @scope {Object} Anything. APE.lock has no use independently - 
  8  * it is purely a locking mechanism.
  9  */
 10 APE.Lock = {
 11 	getLock : function() {
 12 		if(this.__ape_lock_has_lock) return false;
 13 		return this.__ape_lock_has_lock = true;
 14 	},
 15 	releaseLock : function() {
 16 		this.__ape_lock_has_lock = false;
 17 	},
 18 	isLocked : function() {
 19 		return (true === this.__ape_lock_has_lock);
 20 	}
 21 };
